30    /// Get profile for a specific browser and version
31    pub fn get_profile(&self, browser: BrowserType, version: u32) -> Option<BrowserProfile> {
32        // Try exact version first
33        if let Some(entry) = self.registry.get_version(browser, version) {
34            return Some(self.load_profile(&entry.profile_fn));
35        }
36
37        // Find nearest compatible version
38        if let Some(nearest_version) = self.registry.find_nearest_compatible(browser, version) {
39            if let Some(entry) = self.registry.get_version(browser, nearest_version) {
40                return Some(self.load_profile(&entry.profile_fn));
41            }
42        }
43
44        None
45    }
46
47    /// Get profile from User-Agent string
48    pub fn get_profile_from_ua(&self, user_agent: &str) -> Option<BrowserProfile> {
49        let info = VersionDetector::detect(user_agent)?;
50        self.get_profile(info.browser, info.version)
51    }
52
53    /// Get profile for latest browser version
54    pub fn get_latest_profile(&self, browser: BrowserType) -> Option<BrowserProfile> {
55        let entry = self.registry.get_latest(browser)?;
56        Some(self.load_profile(&entry.profile_fn))
57    }

186    /// Get versions with specific capability
187    pub fn get_versions_with_capability(&self, browser: BrowserType, capability: &str) -> Vec<u32> {
188        self.registry
189            .get_with_feature(browser, capability)
190            .into_iter()
191            .map(|(v, _)| *v)
192            .collect()
193    }
194
195    /// Check if a version supports a capability
196    pub fn supports_capability(
197        &self,
198        browser: BrowserType,
199        version: u32,
200        capability: &str,
201    ) -> bool {
202        self.registry
203            .get_version(browser, version)
204            .map(|entry| match capability {
205                "ech" => entry.ech_support,
206                "http3" => entry.http3_support,
207                "psk" => entry.psk_support,
208                "early_data" => entry.early_data_support,
209                "pq" => entry.pq_support,
210                _ => false,
211            })
212            .unwrap_or(false)
213    }

215    /// Get version information
216    pub fn get_version_info(&self, browser: BrowserType, version: u32) -> Option<String> {
217        self.registry.get_version(browser, version).map(|entry| {
218            format!(
219                "{} v{}: {} | ECH: {} | HTTP3: {} | PSK: {} | 0-RTT: {} | PQ: {}",
220                match browser {
221                    BrowserType::Chrome => "Chrome",
222                    BrowserType::Firefox => "Firefox",
223                    BrowserType::Safari => "Safari",
224                    BrowserType::Edge => "Edge",
225                    BrowserType::Opera => "Opera",
226                },
227                entry.version,
228                entry.release_date,
229                if entry.ech_support { "✓" } else { "✗" },
230                if entry.http3_support { "✓" } else { "✗" },
231                if entry.psk_support { "✓" } else { "✗" },
232                if entry.early_data_support {
233                    "✓"
234                } else {
235                    "✗"
236                },
237                if entry.pq_support { "✓" } else { "✗" },
238            )
239        })
240    }
