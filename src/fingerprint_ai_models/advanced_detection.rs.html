39/// Advanced statistical tests for AI detection

42impl AdvancedStatistics {
43    /// Benford's Law analysis for pixel/frequency distributions
44    ///
45    /// Natural images follow Benford's Law for leading digits in measurements.
46    /// AI-generated content often violates this.
47    pub fn benfords_law_test(values: &[u32]) -> f64 {
48        if values.is_empty() {
49            return 0.5;
50        }
51
52        // Expected frequencies for leading digits (Benford's Law)
53        let expected = [
54            0.301, // 1
55            0.176, // 2
56            0.125, // 3
57            0.097, // 4
58            0.079, // 5
59            0.067, // 6
60            0.058, // 7
61            0.051, // 8
62            0.046, // 9
63        ];
64
65        // Count leading digits
66        let mut counts = [0; 9];
67        for &value in values {
68            if value > 0 {
69                let first_digit = Self::get_leading_digit(value);
70                if (1..=9).contains(&first_digit) {
71                    counts[first_digit - 1] += 1;
72                }
73            }
74        }
75
76        // Calculate observed frequencies
77        let total: usize = counts.iter().sum();
78        if total == 0 {
79            return 0.5;
80        }
81
82        let observed: Vec<f64> = counts.iter().map(|&c| c as f64 / total as f64).collect();
83
84        // Chi-square test
85        let chi_square: f64 = expected
86            .iter()
87            .zip(observed.iter())
88            .map(|(&exp, &obs)| {
89                let diff = obs - exp;
90                (diff * diff) / exp
91            })
92            .sum();
93
94        // Normalize chi-square to 0-1 score
95        // Higher chi-square = more deviation from Benford's Law = more AI-like
96        // Chi-square critical value at 0.05 significance for 8 df is ~15.5
97
98        (chi_square / 15.5).min(1.0)
99    }
100
101    /// Get leading (first non-zero) digit
102    fn get_leading_digit(mut n: u32) -> usize {
103        while n >= 10 {
104            n /= 10;
105        }
106        n as usize
107    }

109    /// Chi-square test for distribution uniformity
110    ///
111    /// Tests if a distribution is too uniform (AI-like) or natural
112    pub fn chi_square_uniformity_test(distribution: &[u32], expected_uniform: bool) -> f64 {
113        if distribution.is_empty() {
114            return 0.5;
115        }
116
117        let total: u32 = distribution.iter().sum();
118        if total == 0 {
119            return 0.5;
120        }
121
122        let expected_per_bin = total as f64 / distribution.len() as f64;
123
124        let chi_square: f64 = distribution
125            .iter()
126            .map(|&observed| {
127                let diff = observed as f64 - expected_per_bin;
128                (diff * diff) / expected_per_bin
129            })
130            .sum();
131
132        // Critical value for typical case (adjusted for bin count)
133        let critical_value = distribution.len() as f64 * 2.0;
134
135        if expected_uniform {
136            // High chi-square means non-uniform (natural)
137            // Low chi-square means uniform (AI-like)
138            1.0 - (chi_square / critical_value).min(1.0)
139        } else {
140            // High chi-square means non-uniform (as expected for natural)
141            (chi_square / critical_value).min(1.0)
142        }
143    }

145    /// Kolmogorov-Smirnov test for distribution similarity
146    pub fn ks_test_naturalness(values: &[f64]) -> f64 {
147        if values.len() < 10 {
148            return 0.5;
149        }
150
151        let mut sorted = values.to_vec();
152        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
153
154        // Compare against expected Gaussian distribution
155        let mean = values.iter().sum::<f64>() / values.len() as f64;
156        let variance =
157            values.iter().map(|&x| (x - mean).powi(2)).sum::<f64>() / values.len() as f64;
158        let std_dev = variance.sqrt();
159
160        if std_dev == 0.0 {
161            return 1.0; // Perfect uniformity = very AI-like
162        }
163
164        // Calculate max distance between empirical and theoretical CDF
165        let mut max_distance = 0.0;
166        for (i, &value) in sorted.iter().enumerate() {
167            let empirical_cdf = (i + 1) as f64 / sorted.len() as f64;
168
169            // Gaussian CDF approximation
170            let z = (value - mean) / std_dev;
171            let theoretical_cdf = Self::normal_cdf(z);
172
173            let distance = (empirical_cdf - theoretical_cdf).abs();
174            if distance > max_distance {
175                max_distance = distance;
176            }
177        }
178
179        // K-S statistic, higher = more different from natural
180        // Typical critical value ~0.3 for moderate sample sizes
181
182        (max_distance / 0.3).min(1.0)
183    }
184
185    /// Approximate normal CDF using error function approximation
186    fn normal_cdf(z: f64) -> f64 {
187        0.5 * (1.0 + Self::erf(z / std::f64::consts::SQRT_2))
188    }
189
190    /// Error function approximation
191    fn erf(x: f64) -> f64 {
192        // Abramowitz and Stegun approximation
193        let a1 = 0.254829592;
194        let a2 = -0.284496736;
195        let a3 = 1.421413741;
196        let a4 = -1.453152027;
197        let a5 = 1.061405429;
198        let p = 0.3275911;
199
200        let sign = if x < 0.0 { -1.0 } else { 1.0 };
201        let x = x.abs();
202
203        let t = 1.0 / (1.0 + p * x);
204        let y = 1.0 - (((((a5 * t + a4) * t) + a3) * t + a2) * t + a1) * t * (-x * x).exp();
205
206        sign * y
207    }
208}

210/// Ensemble detector combining multiple algorithms
211pub struct EnsembleDetector;
212
213impl EnsembleDetector {
214    /// Combine multiple detection scores using weighted voting
215    pub fn combine_scores(scores: &HashMap<String, f64>, weights: &HashMap<String, f64>) -> f64 {
216        let mut weighted_sum = 0.0;
217        let mut total_weight = 0.0;
218
219        for (algorithm, &score) in scores {
220            let weight = weights.get(algorithm).copied().unwrap_or(1.0);
221            weighted_sum += score * weight;
222            total_weight += weight;
223        }
224
225        if total_weight > 0.0 {
226            weighted_sum / total_weight
227        } else {
228            0.5
229        }
230    }
231
232    /// Generate explanation for detection
233    pub fn explain_detection(
234        algorithm_results: &HashMap<String, f64>,
235        threshold: f64,
236    ) -> DetectionExplanation {
237        let mut factors = Vec::new();
238
239        // Analyze each algorithm result
240        for (algorithm, &score) in algorithm_results {
241            let description = if score > threshold {
242                format!(
243                    "{} indicates strong AI patterns (score: {:.2})",
244                    algorithm, score
245                )
246            } else {
247                format!("{} shows natural patterns (score: {:.2})", algorithm, score)
248            };
249
250            factors.push((algorithm.clone(), score, description));
251        }
252
253        // Sort by score (highest first)
254        factors.sort_by(|a, b| b.1.partial_cmp(&a.1).unwrap_or(std::cmp::Ordering::Equal));
255
256        // Determine primary reason
257        let primary_reason = if let Some((algo, score, _)) = factors.first() {
258            if *score > threshold {
259                format!(
260                    "Primary indicator: {} (confidence: {:.1}%)",
261                    algo,
262                    score * 100.0
263                )
264            } else {
265                "No strong AI indicators detected".to_string()
266            }
267        } else {
268            "Insufficient data for analysis".to_string()
269        };
270
271        // Calculate explanation confidence based on agreement between algorithms
272        let high_scores = factors
273            .iter()
274            .filter(|(_, score, _)| *score > threshold)
275            .count();
276        let low_scores = factors
277            .iter()
278            .filter(|(_, score, _)| *score <= threshold)
279            .count();
280        let total = factors.len();
281
282        let agreement = if total > 0 {
283            high_scores.max(low_scores) as f64 / total as f64
284        } else {
285            0.5
286        };
287
288        DetectionExplanation {
289            primary_reason,
290            factors,
291            explanation_confidence: agreement,
292        }
293    }

295    /// Calibrate confidence scores to realistic ranges
296    pub fn calibrate_confidence(raw_score: f64, modality: &str) -> f64 {
297        // Different modalities have different baseline accuracies
298        let (min_conf, max_conf) = match modality {
299            "image" => (0.6, 0.95),  // Images are most reliable
300            "text" => (0.5, 0.85),   // Text is moderately reliable
301            "audio" => (0.55, 0.90), // Audio is good
302            "video" => (0.6, 0.92),  // Video is quite reliable
303            _ => (0.5, 0.9),
304        };
305
306        // Map raw score to calibrated range
307        min_conf + (max_conf - min_conf) * raw_score
308    }
