13    pub is_ai_generated: bool,
14
15    /// Confidence score (0.0 - 1.0)
16    pub confidence: f32,

18    /// Perplexity score (lower = more AI-like)

20
21    /// Burstiness score (lower = more AI-like)
22    pub burstiness: f32,
23
24    /// Vocabulary richness score
25    pub vocabulary_richness: f32,
26
27    /// Model attribution probabilities
28    pub model_probabilities: HashMap<String, f32>,
29
30    /// Detected patterns
31    pub patterns: Vec<DetectedPattern>,

33    /// Analysis metadata
34    pub metadata: ContentMetadata,
35}

103/// Analyze text to detect AI-generated content
104///
105/// # Arguments
106///
107/// * `text` - The text content to analyze
108///
109/// # Returns
110///
111/// `ContentFingerprint` containing detection results and analysis
112///
113/// # Example
114///
115/// ```rust
116/// use fingerprint_ai_models::content_detection::detect_ai_content;
117///
118/// let text = "Artificial intelligence has revolutionized numerous sectors...";
119/// let result = detect_ai_content(text);
120///
121/// if result.is_ai_generated {
122///     println!("Detected AI-generated content with {:.2}% confidence",
123///              result.confidence * 100.0);
124/// }
125/// ```
126pub fn detect_ai_content(text: &str) -> ContentFingerprint {
127    // Extract metadata
128    let metadata = analyze_metadata(text);
129
130    // Calculate metrics
131    let perplexity = calculate_perplexity(text, &metadata);
132    let burstiness = calculate_burstiness(text, &metadata);
133    let vocab_richness = calculate_vocabulary_richness(text, &metadata);
134
135    // Detect patterns
136    let patterns = detect_patterns(text, &metadata);
137
138    // Calculate model probabilities
139    let model_probs = attribute_to_models(text, perplexity, burstiness, &patterns);
140
141    // Calculate overall AI likelihood
142    let ai_score = calculate_ai_score(perplexity, burstiness, vocab_richness, &patterns);
143
144    ContentFingerprint {
145        is_ai_generated: ai_score > 0.7,
146        confidence: ai_score,
147        perplexity,
148        burstiness,
149        vocabulary_richness: vocab_richness,
150        model_probabilities: model_probs,
151        patterns,
152        metadata,
153    }
154}

156/// Analyze basic text metadata
157fn analyze_metadata(text: &str) -> ContentMetadata {
158    let char_count = text.chars().count();
159    let words: Vec<&str> = text.split_whitespace().collect();
160    let word_count = words.len();
161
162    // Simple sentence detection (naive approach - can be improved)
163    let sentence_count = text.matches(['.', '!', '?']).count().max(1);
164
165    let avg_sentence_length = word_count as f32 / sentence_count as f32;
166
167    // Count unique words
168    let mut unique_words_set = std::collections::HashSet::new();
169    for word in &words {
170        unique_words_set.insert(word.to_lowercase());
171    }
172    let unique_words = unique_words_set.len();
173
174    ContentMetadata {
175        char_count,
176        sentence_count,
177        word_count,
178        avg_sentence_length,
179        unique_words,
180        analyzed_at: None,
181    }
182}

184/// Calculate perplexity score (simplified version)
185/// Lower perplexity = more predictable = more AI-like
186fn calculate_perplexity(text: &str, metadata: &ContentMetadata) -> f32 {
187    // Simplified perplexity calculation based on word predictability
188    let words: Vec<&str> = text.split_whitespace().collect();
189
190    if words.is_empty() {
191        return 1.0;
192    }
193
194    // Calculate bigram entropy (simplified)
195    let mut bigram_counts: HashMap<(&str, &str), u32> = HashMap::new();
196    let mut unigram_counts: HashMap<&str, u32> = HashMap::new();
197
198    for i in 0..words.len() {
199        *unigram_counts.entry(words[i]).or_insert(0) += 1;
200
201        if i + 1 < words.len() {
202            *bigram_counts.entry((words[i], words[i + 1])).or_insert(0) += 1;
203        }
204    }
205
206    // Normalize by unique words to get perplexity estimate
207    let vocab_size = unigram_counts.len() as f32;
208    let perplexity = vocab_size / (metadata.word_count as f32 + 1.0);
209
210    perplexity.clamp(0.0, 1.0)
211}

213/// Calculate burstiness (sentence length variance)
214/// Lower burstiness = more uniform = more AI-like
215fn calculate_burstiness(text: &str, _metadata: &ContentMetadata) -> f32 {
216    // Split into sentences
217    let sentences: Vec<&str> = text
218        .split(['.', '!', '?'])
219        .filter(|s| !s.trim().is_empty())
220        .collect();
221
222    if sentences.len() < 2 {
223        return 0.5;
224    }
225
226    // Calculate sentence lengths
227    let lengths: Vec<usize> = sentences
228        .iter()
229        .map(|s| s.split_whitespace().count())
230        .collect();
231
232    // Calculate mean
233    let mean = lengths.iter().sum::<usize>() as f32 / lengths.len() as f32;
234
235    // Calculate variance
236    let variance: f32 = lengths
237        .iter()
238        .map(|&l| {
239            let diff = l as f32 - mean;
240            diff * diff
241        })
242        .sum::<f32>()
243        / lengths.len() as f32;
244
245    let std_dev = variance.sqrt();
246
247    // Normalize burstiness (higher variance = more bursty = more human-like)
248    if mean > 0.0 {
249        (std_dev / mean).min(1.0)
250    } else {
251        0.0
252    }
253}

255/// Calculate vocabulary richness (type-token ratio)
256fn calculate_vocabulary_richness(_text: &str, metadata: &ContentMetadata) -> f32 {
257    if metadata.word_count == 0 {
258        return 0.0;
259    }
260
261    // Type-token ratio: unique words / total words
262    metadata.unique_words as f32 / metadata.word_count as f32
263}

265/// Detect AI-specific patterns in text
266fn detect_patterns(text: &str, _metadata: &ContentMetadata) -> Vec<DetectedPattern> {
267    let mut patterns = Vec::new();
268    let text_lower = text.to_lowercase();
269
270    // Check for AI-characteristic phrases
271    let ai_phrases = [
272        "it's important to note",
273        "it's worth noting",
274        "delve into",
275        "navigate the landscape",
276        "in today's digital age",
277        "in conclusion",
278        "furthermore",
279        "moreover",
280        "additionally",
281        "it is essential",
282    ];
283
284    for phrase in &ai_phrases {
285        if text_lower.contains(phrase) {
286            patterns.push(DetectedPattern {
287                pattern_type: PatternType::AiPhrases,
288                description: format!("Found AI-characteristic phrase: '{}'", phrase),
289                confidence: 0.6,
290                location: vec![],
291            });
292        }
293    }
294
295    // Check for repetitive structure
296    let sentences: Vec<&str> = text
297        .split(['.', '!', '?'])
298        .filter(|s| !s.trim().is_empty())
299        .collect();
300
301    if sentences.len() >= 3 {
302        let mut similar_starts = 0;
303        for i in 0..sentences.len() - 1 {
304            let s1_words: Vec<&str> = sentences[i].split_whitespace().take(3).collect();
305            let s2_words: Vec<&str> = sentences[i + 1].split_whitespace().take(3).collect();
306
307            if s1_words.len() >= 2 && s2_words.len() >= 2 && s1_words[0] == s2_words[0] {
308                similar_starts += 1;
309            }
310        }
311
312        if similar_starts > sentences.len() / 3 {
313            patterns.push(DetectedPattern {
314                pattern_type: PatternType::RepetitiveStructure,
315                description: "Repetitive sentence structure detected".to_string(),
316                confidence: 0.7,
317                location: vec![],
318            });
319        }
320    }
321
322    // Check for overly formal language
323    let formal_indicators = text_lower.matches("therefore").count()
324        + text_lower.matches("thus").count()
325        + text_lower.matches("consequently").count()
326        + text_lower.matches("furthermore").count();
327
328    if formal_indicators > text.len() / 500 {
329        patterns.push(DetectedPattern {
330            pattern_type: PatternType::FormalLanguage,
331            description: "Overly formal language detected".to_string(),
332            confidence: 0.5,
333            location: vec![],
334        });
335    }
336
337    patterns
338}

340/// Attribute content to specific AI models
341fn attribute_to_models(
342    text: &str,
343    perplexity: f32,
344    burstiness: f32,
345    patterns: &[DetectedPattern],
346) -> HashMap<String, f32> {
347    let mut probabilities = HashMap::new();
348
349    let text_lower = text.to_lowercase();
350
351    // GPT models tend to be more verbose and use certain phrases
352    let gpt_score =
353        if text_lower.contains("delve into") || text_lower.contains("it's important to note") {
354            0.4 + (1.0 - perplexity) * 0.3
355        } else {
356            (1.0 - perplexity) * 0.3
357        };
358    probabilities.insert("gpt".to_string(), gpt_score.clamp(0.0, 1.0));
359
360    // Claude tends to be more structured and formal
361    let claude_score = if patterns
362        .iter()
363        .any(|p| p.pattern_type == PatternType::FormalLanguage)
364    {
365        0.35 + (1.0 - burstiness) * 0.25
366    } else {
367        (1.0 - burstiness) * 0.25
368    };
369    probabilities.insert("claude".to_string(), claude_score.clamp(0.0, 1.0));
370
371    // Gemini tends to be more concise
372    let gemini_score = if text.len() < 500 && perplexity < 0.3 {
373        0.3 + (1.0 - perplexity) * 0.2
374    } else {
375        (1.0 - perplexity) * 0.2
376    };
377    probabilities.insert("gemini".to_string(), gemini_score.clamp(0.0, 1.0));
378
379    // Normalize probabilities
380    let total: f32 = probabilities.values().sum();
381    if total > 0.0 {
382        for value in probabilities.values_mut() {
383            *value /= total;
384        }
385    }
386
387    probabilities
388}

390/// Calculate overall AI likelihood score
391fn calculate_ai_score(
392    perplexity: f32,
393    burstiness: f32,
394    vocab_richness: f32,
395    patterns: &[DetectedPattern],
396) -> f32 {
397    // Weight different factors
398    let perplexity_weight = 0.3;
399    let burstiness_weight = 0.3;
400    let vocab_weight = 0.2;
401    let pattern_weight = 0.2;
402
403    // Low perplexity = high AI likelihood
404    let perplexity_score = 1.0 - perplexity;
405
406    // Low burstiness = high AI likelihood
407    let burstiness_score = 1.0 - burstiness;
408
409    // Moderate vocab richness is typical for AI
410    let vocab_score = if vocab_richness > 0.3 && vocab_richness < 0.6 {
411        0.7
412    } else {
413        0.3
414    };
415
416    // Pattern detection score
417    let pattern_score = if patterns.is_empty() {
418        0.0
419    } else {
420        let avg_confidence: f32 =
421            patterns.iter().map(|p| p.confidence).sum::<f32>() / patterns.len() as f32;
422        avg_confidence
423    };
424
425    let total_score = perplexity_score * perplexity_weight
426        + burstiness_score * burstiness_weight
427        + vocab_score * vocab_weight
428        + pattern_score * pattern_weight;
429
430    total_score.clamp(0.0, 1.0)
431}
