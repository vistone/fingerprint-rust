1/// Navigator API 噪声注入器
2pub struct NavigatorNoiseInjector {

6impl NavigatorNoiseInjector {
7    pub fn new(seed: u64) -> Self {

11    /// to硬件concurrent数添加噪声
12    pub fn add_hardware_concurrency_noise(&self, cores: u32) -> u32 {
13        use rand::{Rng, SeedableRng};
14        use rand_chacha::ChaCha8Rng;
15
16        let mut rng = ChaCha8Rng::seed_from_u64(self.seed);
17
18        // 有 20% of概率修改核心数 ±1
19        if rng.gen::<f64>() < 0.2 {
20            let noise = if rng.gen::<bool>() { 1 } else { -1 };
21            (cores as i32 + noise).max(1) as u32
22        } else {
23            cores
24        }
25    }
26
27    /// to设备memory添加噪声

33
34        // 有 15% of概率修改memory ±1 GB
35        if rng.gen::<f64>() < 0.15 {
36            let noise = if rng.gen::<bool>() { 1 } else { -1 };
37            (memory_gb as i32 + noise).max(1) as u32
38        } else {
