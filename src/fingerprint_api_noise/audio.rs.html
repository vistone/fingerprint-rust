1/// AudioContext fingerprint噪声
2pub struct AudioNoiseInjector {
3    seed: u64,
4}
5
6impl AudioNoiseInjector {
7    pub fn new(seed: u64) -> Self {
8        Self { seed }
9    }
10
11    /// toaudiodata添加噪声
12    pub fn add_audio_noise(&self, samples: &[f32]) -> Vec<f32> {
13        use rand::{Rng, SeedableRng};
14        use rand_chacha::ChaCha8Rng;
15
16        let mut rng = ChaCha8Rng::seed_from_u64(self.seed);
17        let mut result = samples.to_vec();
18
19        // 对每个sample添加微小of噪声
20        for sample in result.iter_mut() {
21            let noise = rng.gen_range(-0.0001..0.0001);
22            *sample += noise;
23        }
24
25        result
26    }
27
28    /// generate Audio fingerprint（带噪声）
29    pub fn audio_fingerprint(&self, samples: &[f32]) -> Vec<u8> {
30        let noisy_samples = self.add_audio_noise(samples);
31
32        // convertto字节用于hash
33        noisy_samples
34            .iter()
35            .flat_map(|&f| f.to_le_bytes())
36            .collect()
37    }
