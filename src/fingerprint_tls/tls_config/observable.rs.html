6use crate::tls_config::extract::extract_signature;

11/// TLS ClientHello 可observecountdata
12/// includingallcan from ClientHello in observe to info
13/// reference：Huginn Net Profiler TlsClientObserved
14#[derive(Debug, Clone, PartialEq)]
15pub struct TlsClientObserved {
16    /// TLS version (stringrepresent, 如 "13", "12")
17    pub version: String,
18    /// Server Name Indication
19    pub sni: Option<String>,
20    /// Application-Layer Protocol Negotiation
21    pub alpn: Option<String>,
22    /// cipher suitelist
23    pub cipher_suites: Vec<u16>,
24    /// extensionlist
25    pub extensions: Vec<u16>,
26    /// signaturealgorithmlist
27    pub signature_algorithms: Vec<u16>,
28    /// elliptic curvelist
29    pub elliptic_curves: Vec<CurveID>,
30}
31
32impl TlsClientObserved {
33    /// from ClientHelloSpec Create可observecountdata
34    pub fn from_spec(spec: &ClientHelloSpec) -> Self {
35        let signature = extract_signature(spec);
36        Self::from_signature(&signature)
37    }
38
39    /// from ClientHelloSignature Create可observecountdata
40    pub fn from_signature(signature: &ClientHelloSignature) -> Self {
41        Self {
42            version: format!("{}", signature.version),
43            sni: signature.sni.clone(),
44            alpn: signature.alpn.clone(),
45            cipher_suites: signature.cipher_suites.clone(),
46            extensions: signature.extensions.clone(),
47            signature_algorithms: signature.signature_algorithms.clone(),
48            elliptic_curves: signature.elliptic_curves.clone(),
49        }
50    }
51
52    /// Getcipher suitecount
53    pub fn cipher_suite_count(&self) -> usize {
54        self.cipher_suites.len()
55    }
56
57    /// Getextensioncount
58    pub fn extension_count(&self) -> usize {
59        self.extensions.len()
60    }
61
62    /// Getsignaturealgorithmcount
63    pub fn signature_algorithm_count(&self) -> usize {
64        self.signature_algorithms.len()
65    }
66
67    /// Checkwhetherincludingspecificextension
68    pub fn has_extension(&self, ext_id: u16) -> bool {
69        self.extensions.contains(&ext_id)
70    }
71
72    /// Checkwhetherincludingspecificcipher suite
73    pub fn has_cipher_suite(&self, suite: u16) -> bool {
74        self.cipher_suites.contains(&suite)
75    }
76}
