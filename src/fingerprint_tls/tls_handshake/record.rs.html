1//! TLS recordlayer (Record Layer)
2//!
3//! TLS recordformat：
4//! ```text
5//! struct {
6//! ContentType type; // 1 byte
7//! ProtocolVersion version; // 2 bytes
8//! uint16 length; // 2 bytes
9//! opaque fragment[length]; // length bytes
10//! } TLSPlaintext;
11//! ```
12
13/// TLS recordtype
14#[derive(Debug, Clone, Copy, PartialEq, Eq)]
15#[repr(u8)]
16pub enum TLSRecordType {
17    ChangeCipherSpec = 20,
18    Alert = 21,
19    Handshake = 22,
20    ApplicationData = 23,
21}
22
23impl TLSRecordType {
24    pub fn as_u8(&self) -> u8 {
25        *self as u8
26    }
27}
28
29/// TLS record
30#[derive(Debug, Clone)]
31pub struct TLSRecord {
32    /// insidecontaintype
33    pub content_type: TLSRecordType,
34    /// protocolversion (usually is TLS 1.0 = 0x0301,  for compatibleproperty)
35    pub version: u16,
36    /// countdatainsidecontain
37    pub fragment: Vec<u8>,
38}
39
40impl TLSRecord {
41    /// Create a new TLS record
42    pub fn new(content_type: TLSRecordType, version: u16, fragment: Vec<u8>) -> Self {
43        Self {
44            content_type,
45            version,
46            fragment,
47        }
48    }
49
50    /// Createhandshakerecord
51    pub fn handshake(version: u16, data: Vec<u8>) -> Self {
52        Self::new(TLSRecordType::Handshake, version, data)
53    }
54
55    /// serialize as bytesstream
56    pub fn to_bytes(&self) -> Vec<u8> {
57        let mut bytes = Vec::new();
58
59        // Content Type (1 byte)
60        bytes.push(self.content_type.as_u8());
61
62        // Version (2 bytes)
63        bytes.extend_from_slice(&self.version.to_be_bytes());
64
65        // Length (2 bytes)
66        let length = self.fragment.len() as u16;
67        bytes.extend_from_slice(&length.to_be_bytes());
68
69        // Fragment
70        bytes.extend_from_slice(&self.fragment);
71
72        bytes
73    }
74
75    /// from bytesstreamParse
76    pub fn from_bytes(data: &[u8]) -> Result<(Self, usize), String> {
77        if data.len() < 5 {
78            return Err("countdatatoo short，unable toParse TLS record".to_string());
79        }
80
81        let content_type = match data[0] {
82            20 => TLSRecordType::ChangeCipherSpec,
83            21 => TLSRecordType::Alert,
84            22 => TLSRecordType::Handshake,
85            23 => TLSRecordType::ApplicationData,
86            _ => return Err(format!("not知inside容type: {}", data[0])),
87        };
88
89        let version = u16::from_be_bytes([data[1], data[2]]);
90        let length = u16::from_be_bytes([data[3], data[4]]) as usize;
91
92        if data.len() < 5 + length {
93            return Err(format!(
94                "countdata不complete，need {} bytes，actualonly {} bytes",
95                5 + length,
96                data.len()
97            ));
98        }
99
100        let fragment = data[5..5 + length].to_vec();
101
102        Ok((Self::new(content_type, version, fragment), 5 + length))
103    }
104}

106#[cfg(test)]
107mod tests {
108    use super::*;
109
110    #[test]
111    fn test_tls_record_serialization() {
112        let data = vec![1, 2, 3, 4, 5];
113        let record = TLSRecord::handshake(0x0303, data.clone());
114
115        let bytes = record.to_bytes();
116
117        // Validateformat
118        assert_eq!(bytes[0], 22); // Handshake
119        assert_eq!(u16::from_be_bytes([bytes[1], bytes[2]]), 0x0303); // TLS 1.2
120        assert_eq!(u16::from_be_bytes([bytes[3], bytes[4]]), 5); // Length
121        assert_eq!(&bytes[5..], &data);
122    }
123
124    #[test]
125    fn test_tls_record_deserialization() {
126        let data = vec![22, 0x03, 0x03, 0x00, 0x05, 1, 2, 3, 4, 5];
127        let (record, consumed) = TLSRecord::from_bytes(&data).unwrap();
128
129        assert_eq!(record.content_type, TLSRecordType::Handshake);
130        assert_eq!(record.version, 0x0303);
131        assert_eq!(record.fragment, vec![1, 2, 3, 4, 5]);
132        assert_eq!(consumed, 10);
133    }
