9/// TCP configuration
10/// for configuring TCP parameters for connections
11#[derive(Debug, Clone, Copy)]
12pub struct TcpProfile {
13    /// Initial Time-To-Live value
14    pub ttl: u8,
15
16    /// Initial window size
17    pub window_size: u16,
18
19    /// MSS (Maximum Segment Size)
20    pub mss: Option<u16>,
21
22    /// Window Scale
23    pub window_scale: Option<u8>,
24}

26impl Default for TcpProfile {
27    fn default() -> Self {
28        Self {
29            ttl: 64,            // Linux default
30            window_size: 64240, // Typical value for Windows
31            mss: None,          // Operating system default
32            window_scale: None, // Operating system default
33        }
34    }

37impl TcpProfile {
38    /// Generates a TCP profile based on the operating system type
39    ///
40    /// Ensures TCP fingerprint matches browser fingerprint (User-Agent)
41    pub fn for_os(os: crate::types::OperatingSystem) -> Self {
42        match os {
43            crate::types::OperatingSystem::Windows10 | crate::types::OperatingSystem::Windows11 => {
44                // Windows: TTL=128, Window Size=64240 (Windows 10/11 typical values)
45                Self {
46                    ttl: 128,
47                    window_size: 64240,
48                    mss: Some(1460),
49                    window_scale: Some(8),
50                }
51            }
52            crate::types::OperatingSystem::MacOS13
53            | crate::types::OperatingSystem::MacOS14
54            | crate::types::OperatingSystem::MacOS15 => {
55                // macOS: TTL=64, Window Size=65535 (macOS typical values)
56                Self {
57                    ttl: 64,
58                    window_size: 65535,
59                    mss: Some(1460),
60                    window_scale: Some(6),
61                }
62            }
63            crate::types::OperatingSystem::Linux
64            | crate::types::OperatingSystem::LinuxUbuntu
65            | crate::types::OperatingSystem::LinuxDebian => {
66                // Linux: TTL=64, Window Size=65535 (Linux typical values)
67                Self {
68                    ttl: 64,
69                    window_size: 65535,
70                    mss: Some(1460),
71                    window_scale: Some(7),
72                }
73            }
74        }
75    }
76
77    /// Generates a TCP profile by inferring the operating system from User-Agent string
78    ///
79    /// This is a core function for unified fingerprint generation, ensuring browser fingerprint and TCP fingerprint are synchronized
80    pub fn from_user_agent(user_agent: &str) -> Self {
81        use crate::types::OperatingSystem;
82
83        // Infer operating system from User-Agent
84        // Note: iPhone/iPad User-Agent contains "Mac OS X", need to check mobile device first
85        let os = if user_agent.contains("iPhone") || user_agent.contains("iPad") {
86            // iOS device: use macOS TCP fingerprint (iOS based on macOS)
87            OperatingSystem::MacOS14
88        } else if user_agent.contains("Windows NT 10.0") {
89            OperatingSystem::Windows10
90        } else if user_agent.contains("Windows NT 11.0") {
91            OperatingSystem::Windows11
92        } else if user_agent.contains("Mac OS X 13")
93            || user_agent.contains("Macintosh; Intel Mac OS X 13")
94        {
95            OperatingSystem::MacOS13
96        } else if user_agent.contains("Mac OS X 14")
97            || user_agent.contains("Macintosh; Intel Mac OS X 14")
98        {
99            OperatingSystem::MacOS14
100        } else if user_agent.contains("Mac OS X 15")
101            || user_agent.contains("Macintosh; Intel Mac OS X 15")
102        {
103            OperatingSystem::MacOS15
104        } else if user_agent.contains("Linux") || user_agent.contains("Android") {
105            OperatingSystem::Linux
106        } else {
107            // Default to Windows (most common browser environment)
108            OperatingSystem::Windows10
109        };
110
111        Self::for_os(os)
112    }
113
114    /// Generates a TCP profile from platform string (e.g., "Windows", "macOS", "Linux")
115    pub fn from_platform(platform: &str) -> Self {
116        use crate::types::OperatingSystem;
117
118        let os = match platform.to_lowercase().as_str() {
119            "windows" | r#""Windows""# => OperatingSystem::Windows10,
120            "macos" | r#""macOS""# => OperatingSystem::MacOS14,
121            "linux" | r#""Linux""# => OperatingSystem::Linux,
122            _ => OperatingSystem::Windows10, // default
123        };
124
125        Self::for_os(os)
126    }
127}

129/// TCP fingerprint
130#[derive(Debug, Clone)]
131pub struct TcpFingerprint {
132    /// Fingerprint ID (based on TCP trait hash)
133    pub id: String,
134
135    /// Time-To-Live value
136    pub ttl: u8,
137
138    /// Transmission window size
139    pub window_size: u16,
140
141    /// MSS (Maximum Segment Size)
142    pub mss: Option<u16>,
143
144    /// Window scale option
145    pub window_scale: Option<u8>,
146
147    /// TCP options string (for p0f compatibility)
148    pub options_str: Option<String>,
149
150    /// metadata
151    pub metadata: FingerprintMetadata,
152}
153
154impl TcpFingerprint {
155    /// Valid MSS range (minimum Ethernet-compatible MTU minus IP+TCP headers)
156    /// Minimum: 536 (RFC 879 minimum), Maximum: 9000 (practical jumbo frame limit)
157    const MIN_MSS: u16 = 536;
158    const MAX_MSS: u16 = 9000; // Conservative jumbo frame MSS limit
159
160    /// Valid window scale range (0-14 per RFC 7323)
161    const MAX_WINDOW_SCALE: u8 = 14;
162
163    /// Create a new TCP fingerprint
164    pub fn new(ttl: u8, window_size: u16) -> Self {
165        let id = Self::calculate_id(ttl, window_size, None, None);
166        Self {
167            id,
168            ttl,
169            window_size,
170            mss: None,
171            window_scale: None,
172            options_str: None,
173            metadata: FingerprintMetadata::new(),
174        }
175    }
176
177    /// Create a TCP fingerprint with validation
178    ///
179    /// Returns an error if parameters are outside valid ranges:
180    /// - MSS: 536-9000 (RFC 879 minimum to practical jumbo frame limit)
181    /// - Window Scale: 0-14 (RFC 7323)
182    ///
183    /// # Example
184    ///
185    /// ```
186    /// use fingerprint_core::tcp::TcpFingerprint;
187    ///
188    /// let fp = TcpFingerprint::new_validated(64, 65535, Some(1460), Some(7));
189    /// assert!(fp.is_ok());
190    ///
191    /// let invalid = TcpFingerprint::new_validated(64, 65535, Some(100), Some(7));
192    /// assert!(invalid.is_err()); // MSS too small
193    /// ```
194    pub fn new_validated(
195        ttl: u8,
196        window_size: u16,
197        mss: Option<u16>,
198        window_scale: Option<u8>,
199    ) -> Result<Self, String> {
200        // Validate MSS if provided
201        if let Some(mss_val) = mss {
202            if mss_val < Self::MIN_MSS {
203                return Err(format!(
204                    "MSS value {} is below minimum {} (RFC 879)",
205                    mss_val,
206                    Self::MIN_MSS
207                ));
208            }
209            if mss_val > Self::MAX_MSS {
210                return Err(format!(
211                    "MSS value {} exceeds maximum {}",
212                    mss_val,
213                    Self::MAX_MSS
214                ));
215            }
216        }
217
218        // Validate window scale if provided
219        if let Some(ws_val) = window_scale {
220            if ws_val > Self::MAX_WINDOW_SCALE {
221                return Err(format!(
222                    "Window scale {} exceeds maximum {} (RFC 7323)",
223                    ws_val,
224                    Self::MAX_WINDOW_SCALE
225                ));
226            }
227        }
228
229        Ok(Self::with_options(ttl, window_size, mss, window_scale))
230    }
231
232    /// Create complete TCP fingerprint (no validation, use new_validated for safe construction)
233    pub fn with_options(
234        ttl: u8,
235        window_size: u16,
236        mss: Option<u16>,
237        window_scale: Option<u8>,
238    ) -> Self {
239        let id = Self::calculate_id(ttl, window_size, mss, window_scale);
240        Self {
241            id,
242            ttl,
243            window_size,
244            mss,
245            window_scale,
246            options_str: None,
247            metadata: FingerprintMetadata::new(),
248        }
249    }
250
251    /// Calculates the fingerprint ID
252    fn calculate_id(
253        ttl: u8,
254        window_size: u16,
255        mss: Option<u16>,
256        window_scale: Option<u8>,
257    ) -> String {
258        use sha2::{Digest, Sha256};
259        let mut hasher = Sha256::new();
260        hasher.update([ttl]);
261        hasher.update(window_size.to_be_bytes());
262        if let Some(mss_val) = mss {
263            hasher.update(mss_val.to_be_bytes());
264        }
265        if let Some(ws_val) = window_scale {
266            hasher.update([ws_val]);
267        }
268        format!("{:x}", hasher.finalize())
269    }
270
271    /// Infers the initial TTL according to common OS defaults
272    pub fn infer_initial_ttl(&self) -> u8 {
273        // Based on current TTL, infer the initial TTL
274        // Common initial TTL values: 64 (Linux), 128 (Windows), 255 (Unix)
275        if self.ttl <= 64 {
276            64
277        } else if self.ttl <= 128 {
278            128
279        } else {
280            255
281        }
282    }
283}

285impl Fingerprint for TcpFingerprint {
286    fn fingerprint_type(&self) -> FingerprintType {
287        FingerprintType::Tcp
288    }
289
290    fn id(&self) -> String {
291        self.id.clone()
292    }
293
294    fn metadata(&self) -> &FingerprintMetadata {
295        &self.metadata
296    }
297
298    fn metadata_mut(&mut self) -> &mut FingerprintMetadata {
299        &mut self.metadata
300    }
301
302    fn hash(&self) -> u64 {
303        use std::collections::hash_map::DefaultHasher;
304        let mut hasher = DefaultHasher::new();
305        self.ttl.hash(&mut hasher);
306        self.window_size.hash(&mut hasher);
307        self.mss.hash(&mut hasher);
308        self.window_scale.hash(&mut hasher);
309        hasher.finish()
310    }
311
312    fn similar_to(&self, other: &dyn Fingerprint) -> bool {
313        if other.fingerprint_type() != FingerprintType::Tcp {
314            return false;
315        }
316
317        // TCP fingerprint similarity judgment: allow certain tolerance
318        // Simplified process here, should actually consider TTL inferred value, Window Size multiple close system etc.
319        self.hash() == other.hash()
320    }
321
322    fn to_string(&self) -> String {
323        format!(
324            "TcpFingerprint(id={}, ttl={}, window={})",
325            self.id, self.ttl, self.window_size
326        )
327    }
328}
