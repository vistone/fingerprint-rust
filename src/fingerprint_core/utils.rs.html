7/// from slice in randomly select anelement (threadsecurity)
8/// use thread_rng() ensurethreadsecurity
9pub fn random_choice<T: Clone>(items: &[T]) -> Option<T> {
10    if items.is_empty() {
11        return None;
12    }
13    let mut rng = rand::thread_rng();
14    let index = rng.gen_range(0..items.len());
15    Some(items[index].clone())
16}

18/// from stringslice in randomly select anelement (threadsecurity)
19pub fn random_choice_string(items: &[&str]) -> Option<String> {
20    random_choice(items).map(|s| s.to_string())
21}

23/// from User-Agent in Extract Chrome versionnumber
24pub fn extract_chrome_version(user_agent: &str) -> String {
25    // find "Chrome/" back面versionnumber
26    if let Some(start) = user_agent.find("Chrome/") {
27        let version_start = start + 7; // "Chrome/".len()
28        if let Some(end) =
29            user_agent[version_start..].find(|c: char| !c.is_ascii_digit() && c != '.')
30        {
31            return user_agent[version_start..version_start + end].to_string();
32        }
33        // Ifnot found to endbitplace, return to stringend尾
34        return user_agent[version_start..]
35            .split_whitespace()
36            .next()
37            .unwrap_or("120")
38            .to_string();
39    }
40    "120".to_string() // defaultversion
41}

43/// from User-Agent in Extractplatforminfo
44pub fn extract_platform(user_agent: &str) -> String {
45    // Extractplatforminfo for Sec-CH-UA-Platform
46    if user_agent.contains("Windows") {
47        return r#""Windows""#.to_string();
48    } else if user_agent.contains("Macintosh") || user_agent.contains("Mac OS X") {
49        return r#""macOS""#.to_string();
50    } else if user_agent.contains("Linux") {
51        return r#""Linux""#.to_string();
52    } else if user_agent.contains("Android") {
53        return r#""Android""#.to_string();
54    } else if user_agent.contains("iPhone") || user_agent.contains("iPad") {
55        return r#""iOS""#.to_string();
56    }
57    r#""Windows""#.to_string() // defaultplatform
58}

60/// from User-Agent in Extractoperating systemtype
61///
62/// for unifiedfingerprintGenerate, ensurebrowserfingerprint and TCP fingerprintsync
63pub fn extract_os_from_user_agent(user_agent: &str) -> crate::types::OperatingSystem {
64    use crate::types::OperatingSystem;
65
66    // Note: iPhone/iPad User-Agent including "Mac OS X", need先Checkmovedevice
67    if user_agent.contains("iPhone") || user_agent.contains("iPad") {
68        // iOS device：use macOS TCP fingerprint (iOS based on macOS)
69        OperatingSystem::MacOS14
70    } else if user_agent.contains("Windows NT 10.0") {
71        OperatingSystem::Windows10
72    } else if user_agent.contains("Windows NT 11.0") {
73        OperatingSystem::Windows11
74    } else if user_agent.contains("Mac OS X 13")
75        || user_agent.contains("Macintosh; Intel Mac OS X 13")
76    {
77        OperatingSystem::MacOS13
78    } else if user_agent.contains("Mac OS X 14")
79        || user_agent.contains("Macintosh; Intel Mac OS X 14")
80    {
81        OperatingSystem::MacOS14
82    } else if user_agent.contains("Mac OS X 15")
83        || user_agent.contains("Macintosh; Intel Mac OS X 15")
84    {
85        OperatingSystem::MacOS15
86    } else if user_agent.contains("Linux") || user_agent.contains("Android") {
87        OperatingSystem::Linux
88    } else {
89        // defaultuse Windows (most commonbrowserenvironment)
90        OperatingSystem::Windows10
91    }
92}

94/// from profile nameinferbrowsertype
95pub fn infer_browser_from_profile_name(profile_name: &str) -> (String, bool) {
96    let name_lower = profile_name.to_lowercase();
97    if name_lower.starts_with("chrome_") {
98        ("chrome".to_string(), false)
99    } else if name_lower.starts_with("firefox_") {
100        ("firefox".to_string(), false)
101    } else if name_lower.starts_with("safari_") {
102        (
103            "safari".to_string(),
104            name_lower.contains("ios") || name_lower.contains("ipad"),
105        )
106    } else if name_lower.starts_with("opera_") {
107        ("opera".to_string(), false)
108    } else if name_lower.contains("ios")
109        || name_lower.contains("android")
110        || name_lower.contains("mobile")
111    {
112        // mobileapplicationfingerprint
113        if name_lower.contains("ios") {
114            ("safari".to_string(), true)
115        } else {
116            ("chrome".to_string(), true)
117        }
118    } else {
119        ("chrome".to_string(), false) // default
120    }
121}

123/// judgewhether as mobile profile
124pub fn is_mobile_profile(profile_name: &str) -> bool {
125    let name = profile_name.to_lowercase();
126    name.contains("ios")
127        || name.contains("android")
128        || name.contains("ipad")
129        || name.contains("mobile")
130}
