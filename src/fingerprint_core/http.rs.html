51impl HttpFingerprint {
52    /// Create a new HTTP fingerprint
53    pub fn new(user_agent: String, headers: HashMap<String, String>) -> Self {
54        let id = Self::calculate_id(&user_agent, &headers);
55        Self {
56            id,
57            user_agent,
58            headers,
59            http2_settings: None,
60            metadata: FingerprintMetadata::new(),
61        }
62    }
63
64    /// Calculatefingerprint ID
65    fn calculate_id(user_agent: &str, headers: &HashMap<String, String>) -> String {
66        use sha2::{Digest, Sha256};
67        let mut hasher = Sha256::new();
68        hasher.update(user_agent.as_bytes());
69
70        // pair headers performsortbackhash
71        let mut header_vec: Vec<_> = headers.iter().collect();
72        header_vec.sort_by_key(|(k, _)| *k);
73        for (k, v) in header_vec {
74            hasher.update(k.as_bytes());
75            hasher.update(v.as_bytes());
76        }
77
78        format!("{:x}", hasher.finalize())
79    }
80
81    /// settings HTTP/2 settings
82    pub fn with_http2_settings(mut self, settings: Http2Settings) -> Self {
83        self.http2_settings = Some(settings);
84        // reCalculate ID (including HTTP/2 settings)
85        self.id = Self::calculate_id(&self.user_agent, &self.headers);
86        self
87    }
88}

90impl Fingerprint for HttpFingerprint {
91    fn fingerprint_type(&self) -> FingerprintType {
92        FingerprintType::Http
93    }
94
95    fn id(&self) -> String {
96        self.id.clone()
97    }
98
99    fn metadata(&self) -> &FingerprintMetadata {
100        &self.metadata
101    }
102
103    fn metadata_mut(&mut self) -> &mut FingerprintMetadata {
104        &mut self.metadata
105    }
106
107    fn hash(&self) -> u64 {
108        use std::collections::hash_map::DefaultHasher;
109        let mut hasher = DefaultHasher::new();
110        self.user_agent.hash(&mut hasher);
111
112        // pair headers performsortbackhash
113        let mut header_vec: Vec<_> = self.headers.iter().collect();
114        header_vec.sort_by_key(|(k, _)| *k);
115        for (k, v) in header_vec {
116            k.hash(&mut hasher);
117            v.hash(&mut hasher);
118        }
119
120        if let Some(ref settings) = self.http2_settings {
121            settings.header_table_size.hash(&mut hasher);
122            settings.enable_push.hash(&mut hasher);
123            settings.max_concurrent_streams.hash(&mut hasher);
124            settings.initial_window_size.hash(&mut hasher);
125            settings.max_frame_size.hash(&mut hasher);
126            settings.max_header_list_size.hash(&mut hasher);
127        }
128
129        hasher.finish()
130    }
131
132    fn similar_to(&self, other: &dyn Fingerprint) -> bool {
133        if other.fingerprint_type() != FingerprintType::Http {
134            return false;
135        }
136
137        // tryconvert to HttpFingerprint
138        // Note: hereneedtypeConvert, butdue to trait limit, wecan onlycomparehashvalue
139        self.hash() == other.hash()
140    }
141
142    fn to_string(&self) -> String {
143        format!("HttpFingerprint(id={}, ua={})", self.id, self.user_agent)
144    }
145}
