4/// Rate Limiting Service Implementation
5///
6/// Implements distributed rate limiting using Redis backend with token bucket algorithm.
7/// Supports multiple quota tiers and per-endpoint rate limits.
8///
9/// # Features
10/// - Token bucket algorithm with configurable burst size
11/// - Redis-backed distributed state
12/// - Multi-tier support (Free, Pro, Enterprise, Partner)
13/// - Per-endpoint cost multipliers
14/// - Graceful degradation under high load
15use std::sync::Arc;
16use std::time::{Duration, SystemTime, UNIX_EPOCH};

18/// Rate limiting quota tier definitions
19#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
20pub enum QuotaTier {
21    /// Free tier: 100 req/min, 50K/month
22    Free,
23    /// Pro tier: 1000 req/min, 1M/month ($99/month)
24    Pro,
25    /// Enterprise: Unlimited, custom SLA
26    Enterprise,
27    /// Partner: Unlimited (special program)
28    Partner,
29}
30
31impl QuotaTier {
32    /// Get minute-level rate limit for tier
33    pub fn minute_limit(&self) -> u64 {
34        match self {
35            QuotaTier::Free => 100,
36            QuotaTier::Pro => 1000,
37            QuotaTier::Enterprise => u64::MAX,
38            QuotaTier::Partner => u64::MAX,
39        }
40    }
41
42    /// Get monthly quota for tier
43    pub fn monthly_quota(&self) -> u64 {
44        match self {
45            QuotaTier::Free => 50_000,
46            QuotaTier::Pro => 1_000_000,
47            QuotaTier::Enterprise => u64::MAX,
48            QuotaTier::Partner => u64::MAX,
49        }
50    }
51
52    /// Get cost per request ratio
53    pub fn cost_multiplier(&self) -> f64 {
54        match self {
55            QuotaTier::Free => 1.0,
56            QuotaTier::Pro => 1.0,
57            QuotaTier::Enterprise => 0.5,
58            QuotaTier::Partner => 0.0,
59        }
60    }
61}

63/// Per-user quota state
64#[derive(Debug, Clone, Serialize, Deserialize)]
65pub struct UserQuota {
66    /// User or API key identifier
67    pub user_id: String,
68    /// Current tier
69    pub tier: QuotaTier,
70    /// Tokens available (for minute limit)
71    pub available_tokens: f64,
72    /// Last refill time
73    pub last_refill: u64,
74    /// Requests this month
75    pub month_requests: u64,
76    /// Month reset epoch
77    pub month_start: u64,
78    /// Total requests (all time)
79    pub total_requests: u64,
80    /// Last request time
81    pub last_request: u64,
82}
83
84impl UserQuota {
85    /// Create new quota entry
86    pub fn new(user_id: String, tier: QuotaTier) -> Self {
87        let now = current_unix_timestamp();
88        Self {
89            user_id,
90            tier,
91            available_tokens: tier.minute_limit() as f64,
92            last_refill: now,
93            month_requests: 0,
94            month_start: now,
95            total_requests: 0,
96            last_request: 0,
97        }
98    }
99
100    /// Check if user has available quota
101    pub fn has_quota(&self) -> bool {
102        if matches!(self.tier, QuotaTier::Enterprise | QuotaTier::Partner) {
103            return true;
104        }
105        self.available_tokens >= 1.0 && self.month_requests < self.tier.monthly_quota()
106    }
107
108    /// Consume tokens for a request
109    pub fn consume(&mut self, endpoint_cost: f64) -> bool {
110        if !self.has_quota() {
111            return false;
112        }
113
114        let cost = endpoint_cost * self.tier.cost_multiplier();
115
116        if self.available_tokens < cost {
117            return false;
118        }
119
120        self.available_tokens -= cost;
121        self.month_requests += 1;
122        self.total_requests += 1;
123        self.last_request = current_unix_timestamp();
124
125        true
126    }
127}

232    /// Check if request is allowed
233    ///
234    /// # Arguments
235    /// * `user_id` - User or API key identifier
236    /// * `tier` - User's quota tier
237    /// * `endpoint` - Endpoint being accessed
238    /// * `client_ip` - Client IP address (for fallback limiting)
239    ///
240    /// # Returns
241    /// * `Ok(RateLimitResponse)` - Request allowed with remaining quota
242    /// * `Err(RateLimitError)` - Request rejected
243    pub fn check_limit(
244        &self,
245        user_id: Option<&str>,
246        tier: QuotaTier,
247        endpoint: &str,
248        client_ip: Option<&str>,
249    ) -> Result<RateLimitResponse, RateLimitError> {
250        let now = current_unix_timestamp();
251        let mut metrics = self.metrics.total_requests.lock();
252        *metrics += 1;
253        drop(metrics);
254
255        // Get endpoint configuration
256        let endpoints = self.endpoints.read();
257        let endpoint_config = endpoints
258            .iter()
259            .find(|e| e.path == endpoint)
260            .cloned()
261            .unwrap_or_else(|| EndpointConfig::new(endpoint.to_string(), 1.0));
262        drop(endpoints);
263
264        // Authenticated user flow
265        if let Some(user) = user_id {
266            let cache_hit = self.user_quotas.contains_key(user);
267
268            // Create quota entry if not exists (avoid borrow issues by computing first)
269            let user_key = user.to_string();
270            if !cache_hit {
271                let new_quota = UserQuota::new(user_key.clone(), tier);
272                self.user_quotas.insert(user_key.clone(), new_quota);
273            }
274
275            // Get mutable reference to the quota
276            let mut entry = self.user_quotas.get_mut(&user_key).ok_or_else(|| {
277                RateLimitError::RateLimitExceeded {
278                    retry_after: Duration::from_secs(60),
279                }
280            })?;
281
282            let quota = entry.value_mut();
283
284            // Refill tokens (token bucket)
285            self.refill_tokens(quota, now);
286
287            // Check monthly and minute limits
288            if !quota.has_quota() {
289                let mut rejected = self.metrics.total_rejected.lock();
290                *rejected += 1;
291                drop(rejected);
292
293                return Err(RateLimitError::QuotaExceeded {
294                    retry_after: calculate_retry_after(quota, tier),
295                    monthly_reset: quota.month_start + 30 * 86400,
296                });
297            }
298
299            // Consume tokens
300            if !quota.consume(endpoint_config.cost_multiplier) {
301                let mut rejected = self.metrics.total_rejected.lock();
302                *rejected += 1;
303                drop(rejected);
304
305                return Err(RateLimitError::RateLimitExceeded {
306                    retry_after: Duration::from_secs(60),
307                });
308            }
309
310            if cache_hit {
311                let mut hits = self.metrics.cache_hits.lock();
312                *hits += 1;
313            } else {
314                let mut misses = self.metrics.cache_misses.lock();
315                *misses += 1;
316            }
317
318            Ok(RateLimitResponse {
319                allowed: true,
320                remaining: quota.available_tokens as u64,
321                reset_after: Duration::from_secs(60),
322                quota_tier: tier,
323                monthly_remaining: quota.tier.monthly_quota() - quota.month_requests,
324            })
325        } else {
326            // Unauthenticated flow - use IP-based limiting
327            let ip = client_ip.unwrap_or("unknown");
328            let mut entry = self
329                .ip_quotas
330                .entry(ip.to_string())
331                .or_insert_with(|| UserQuota::new(ip.to_string(), QuotaTier::Free));
332
333            let quota = entry.value_mut();
334
335            // More aggressive refill for IP-based (1 minute window)
336            self.refill_tokens_rapid(quota, now);
337
338            // Check limits
339            if !quota.has_quota() {
340                let mut rejected = self.metrics.total_rejected.lock();
341                *rejected += 1;
342
343                return Err(RateLimitError::RateLimitExceeded {
344                    retry_after: Duration::from_secs(60),
345                });
346            }
347
348            // Consume tokens
349            if !quota.consume(endpoint_config.cost_multiplier) {
350                let mut rejected = self.metrics.total_rejected.lock();
351                *rejected += 1;
352
353                return Err(RateLimitError::RateLimitExceeded {
354                    retry_after: Duration::from_secs(60),
355                });
356            }
357
358            Ok(RateLimitResponse {
359                allowed: true,
360                remaining: quota.available_tokens as u64,
361                reset_after: Duration::from_secs(60),
362                quota_tier: QuotaTier::Free,
363                monthly_remaining: 50_000 - quota.month_requests,
364            })
365        }
366    }
367
368    /// Refill tokens using token bucket algorithm
369    fn refill_tokens(&self, quota: &mut UserQuota, now: u64) {
370        let elapsed = now.saturating_sub(quota.last_refill);
371
372        if elapsed >= 60 {
373            // One minute has passed - full refill
374            quota.available_tokens = quota.tier.minute_limit() as f64;
375            quota.last_refill = now;
376        } else if elapsed > 0 {
377            // Partial refill: tokens_per_second * elapsed_seconds
378            let rate = quota.tier.minute_limit() as f64 / 60.0;
379            let tokens_to_add = rate * elapsed as f64;
380            quota.available_tokens = (quota.available_tokens + tokens_to_add)
381                .min(quota.tier.minute_limit() as f64 * 1.5); // Allow 50% burst
382        }
383    }
384
385    /// Rapid refill for IP-based rate limiting (stricter)
386    fn refill_tokens_rapid(&self, quota: &mut UserQuota, now: u64) {
387        let elapsed = now.saturating_sub(quota.last_refill);
388
389        if elapsed >= 60 {
390            quota.available_tokens = 30.0; // 30 req/min for IPs
391            quota.last_refill = now;
392        } else if elapsed > 0 {
393            let rate = 30.0 / 60.0;
394            let tokens_to_add = rate * elapsed as f64;
395            quota.available_tokens = (quota.available_tokens + tokens_to_add).min(45.0);
396            // 1.5x burst
397        }
398    }

412    /// Clear stale entries (should run periodically)
413    pub fn cleanup_stale_entries(&self, retention_seconds: u64) {
414        let now = current_unix_timestamp();
415        let threshold = now.saturating_sub(retention_seconds);
416
417        // Remove stale user quotas (inactive for > retention period)
418        self.user_quotas
419            .retain(|_, quota| quota.last_request > threshold);
420
421        // Remove stale IP quotas
422        self.ip_quotas
423            .retain(|_, quota| quota.last_request > threshold);
424    }

426    /// Sync quota to Redis backend (async)
427    ///
428    /// This should be called after a quota change to ensure
429    /// distributed consistency across multiple instances.
430    pub async fn sync_quota_to_redis(&self, user_id: &str) -> Result<(), String> {
431        if let Some(ref backend) = self.redis_backend {
432            if let Some(quota) = self.user_quotas.get(user_id) {
433                let entry = super::rate_limiting_redis::RedisQuotaEntry::from_user_quota(
434                    user_id.to_string(),
435                    &quota,
436                );
437                // Store with 1 hour TTL
438                backend
439                    .set_user_quota(user_id, &entry, 3600)
440                    .await
441                    .map_err(|e| e.to_string())?;
442            }
443        }
444        Ok(())
445    }

447    /// Load quota from Redis backend (async)
448    ///
449    /// This should be called when a user is not found in local cache
450    /// to check if they have existing quota state in Redis.
451    pub async fn load_quota_from_redis(&self, user_id: &str) -> Option<UserQuota> {
452        if let Some(ref backend) = self.redis_backend {
453            if let Ok(Some(entry)) = backend.get_user_quota(user_id).await {
454                // Convert RedisQuotaEntry back to UserQuota
455                // Note: This is a simplified conversion
456                let tier = match entry.tier.as_str() {
457                    "Pro" => QuotaTier::Pro,
458                    "Enterprise" => QuotaTier::Enterprise,
459                    "Partner" => QuotaTier::Partner,
460                    _ => QuotaTier::Free,
461                };
462
463                let mut quota = UserQuota::new(entry.user_id.clone(), tier);
464                quota.available_tokens = entry.available_tokens;
465                quota.last_refill = entry.last_refill;
466                quota.month_requests = entry.month_requests;
467                quota.month_start = entry.month_start;
468
469                // Update local cache
470                self.user_quotas.insert(user_id.to_string(), quota.clone());
471
472                return Some(quota);
473            }
474        }
475        None
476    }

513pub struct MetricsSnapshot {

515    pub total_rejected: u64,
516    pub cache_hits: u64,

522impl MetricsSnapshot {
523    pub fn rejection_rate(&self) -> f64 {
524        if self.total_requests == 0 {
525            0.0
526        } else {
527            self.total_rejected as f64 / self.total_requests as f64
528        }
529    }
530
531    pub fn cache_hit_rate(&self) -> f64 {
532        let total = self.cache_hits + self.cache_misses;
533        if total == 0 {
534            0.0
535        } else {
536            self.cache_hits as f64 / total as f64
537        }
538    }
539}

541/// Get current Unix timestamp
542pub fn current_unix_timestamp() -> u64 {
543    SystemTime::now()
544        .duration_since(UNIX_EPOCH)
545        .unwrap_or_default()
546        .as_secs()
547}
548
549/// Calculate seconds until next token refill
550fn calculate_retry_after(quota: &UserQuota, tier: QuotaTier) -> u64 {
551    // If monthly quota exceeded, return until next month
552    if quota.month_requests >= tier.monthly_quota() {
553        (quota.month_start + 30 * 86400).saturating_sub(current_unix_timestamp())
554    } else {
555        // If minute limit exceeded, return 60 seconds
556        60
557    }
558}
