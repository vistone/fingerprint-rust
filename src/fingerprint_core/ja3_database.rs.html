9/// Browser information from JA3 match
10#[derive(Debug, Clone, PartialEq)]
11pub struct BrowserMatch {
12    /// Browser name (e.g., "Chrome", "Firefox")
13    pub browser: String,
14    /// Specific version (e.g., "136.0.6778.108")
15    pub version: String,
16    /// Match confidence (0.0 - 1.0)
17    pub confidence: f64,
18    /// Additional notes
19    pub notes: Option<String>,
20}
21
22/// JA3 Fingerprint Database
23pub struct JA3Database {
24    /// Map of JA3 hash -> Browser information
25    fingerprints: HashMap<String, Vec<BrowserMatch>>,
26}

176    /// Add a fingerprint to the database
177    fn add_fingerprint(&mut self, ja3: &str, match_info: BrowserMatch) {
178        self.fingerprints
179            .entry(ja3.to_string())
180            .or_default()
181            .push(match_info);
182    }
183
184    /// Match a JA3 hash against the database
185    pub fn match_ja3(&self, ja3: &str) -> Option<BrowserMatch> {
186        // Exact match
187        if let Some(matches) = self.fingerprints.get(ja3) {
188            // Return the first (highest priority) match
189            return matches.first().cloned();
190        }
191
192        // Try fuzzy matching (for JA3 strings with minor variations)
193        self.fuzzy_match(ja3)
194    }
195
196    /// Fuzzy match for JA3 fingerprints with minor variations
197    fn fuzzy_match(&self, ja3: &str) -> Option<BrowserMatch> {
198        // If JA3 is already a hash (32 chars hex), no fuzzy matching
199        if ja3.len() == 32 && ja3.chars().all(|c| c.is_ascii_hexdigit()) {
200            return None;
201        }
202
203        // For JA3 strings (format: version,ciphers,extensions,curves,formats)
204        // Compare against stored JA3 strings using GREASE-aware similarity
205        let mut candidates: Vec<(f64, BrowserMatch)> = Vec::new();
206
207        for (stored_ja3, matches) in &self.fingerprints {
208            // Only compare against JA3 strings (not hashes)
209            if stored_ja3.len() == 32 {
210                continue;
211            }
212
213            let score = if grease::ja3_equal_ignore_grease(ja3, stored_ja3) {
214                // Exact match after GREASE removal
215                0.95
216            } else {
217                // Use GREASE-normalized similarity for cross-session matching
218                grease::ja3_similarity(ja3, stored_ja3)
219            };
220
221            // Only consider matches with 80%+ similarity
222            if score >= 0.80 {
223                if let Some(match_info) = matches.first() {
224                    let mut fuzzy_match = match_info.clone();
225                    fuzzy_match.confidence *= score;
226                    candidates.push((score, fuzzy_match));
227                }
228            }
229        }
230
231        // Sort by score (highest first) and return the best match
232        if !candidates.is_empty() {
233            candidates.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
234            Some(candidates[0].1.clone())
235        } else {
236            None
237        }
238    }

240    /// Get all known fingerprints
241    pub fn get_all_fingerprints(&self) -> Vec<(&str, &BrowserMatch)> {
242        let mut result = Vec::new();
243        for (ja3, matches) in &self.fingerprints {
244            for match_info in matches {
245                result.push((ja3.as_str(), match_info));
246            }
247        }
248        result
249    }
250
251    /// Get fingerprint count
252    pub fn count(&self) -> usize {
253        self.fingerprints.values().map(|v| v.len()).sum()
254    }

358        let db = JA3Database::new();
359
360        // Same Chrome JA3 with different GREASE values from different sessions
361        // Session 1: GREASE 0x0a0a
362        let chrome_session1_grease_0a0a = "771,4865-4866-4867-49195-49199-49196-49200-52393-52392-49171-49172-156-157-47-53,0-23-65281-10-11-35-16-5-13-18-51-45-43-27-21-0a0a,29-23-24,0";
363
364        // Session 2: GREASE 0x1a1a
365        let chrome_session2_grease_1a1a = "771,4865-4866-4867-49195-49199-49196-49200-52393-52392-49171-49172-156-157-47-53,0-23-65281-10-11-35-16-5-13-18-51-45-43-27-21-1a1a,29-23-24,0";
366
367        // Session 3: GREASE 0x2a2a
368        let chrome_session3_grease_2a2a = "771,4865-4866-4867-49195-49199-49196-49200-52393-52392-49171-49172-156-157-47-53,0-23-65281-10-11-35-16-5-13-18-51-45-43-27-21-2a2a,29-23-24,0";
369
370        let result1 = db.match_ja3(chrome_session1_grease_0a0a);
371        let result2 = db.match_ja3(chrome_session2_grease_1a1a);
372        let result3 = db.match_ja3(chrome_session3_grease_2a2a);
373
374        // All three should match successfully
375        assert!(result1.is_some(), "Session 1 should match");
376        assert!(result2.is_some(), "Session 2 should match");
377        assert!(result3.is_some(), "Session 3 should match");
378
379        // All three should identify as the same browser (or at least not fail)
380        if let (Some(m1), Some(m2), Some(m3)) = (result1, result2, result3) {
381            // Should all be confident matches
382            assert!(m1.confidence >= 0.70, "Session 1 confidence >= 70%");
383            assert!(m2.confidence >= 0.70, "Session 2 confidence >= 70%");
384            assert!(m3.confidence >= 0.70, "Session 3 confidence >= 70%");
385
386            // All should match to a browser (Chrome, Firefox, or similar family)
387            // The key is that GREASE differences don't prevent matching
388            println!(
389                "Chrome Session 1 (0x0a0a): {} {} ({}%)",
390                m1.browser,
391                m1.version,
392                m1.confidence * 100.0
