1//! TLS ClientHello Signature module
2//!
3//! Provides TLS ClientHello signature extraction and comparison features
4//! Reference: Huginn Net Signature structure design

6use crate::dicttls::supported_groups::CurveID;
7use crate::fingerprint::{Fingerprint, FingerprintType};
8use crate::grease::{filter_grease_values, is_grease_value};
9use crate::metadata::FingerprintMetadata;
10use crate::stable_hash::StableHashBuilder;
11use crate::version::TlsVersion;
12use sha2::{Digest, Sha256};

14/// TLS ClientHello signature
15/// Extracts all key information from ClientHello message
16#[derive(Debug, Clone, PartialEq)]
17pub struct ClientHelloSignature {
18    /// Fingerprint ID (based on JA4 hash or signature trait hash)
19    pub id: String,
20
21    /// TLS version
22    pub version: TlsVersion,
23    /// Cipher suite list (including GREASE)
24    pub cipher_suites: Vec<u16>,
25    /// Extension list (including GREASE)
26    pub extensions: Vec<u16>,
27    /// Elliptic curve list
28    pub elliptic_curves: Vec<CurveID>,
29    /// Elliptic curve point format
30    pub elliptic_curve_point_formats: Vec<u8>,
31    /// Signature algorithm list
32    pub signature_algorithms: Vec<u16>,
33    /// Server Name Indication
34    pub sni: Option<String>,
35    /// Application-Layer Protocol Negotiation
36    pub alpn: Option<String>,
37
38    /// metadata
39    pub metadata: FingerprintMetadata,
40}
41
42impl ClientHelloSignature {
43    /// Creates a new signature
44    pub fn new() -> Self {
45        let mut sig = Self {
46            id: String::new(),
47            version: TlsVersion::V1_2, // default TLS 1.2
48            cipher_suites: Vec::new(),
49            extensions: Vec::new(),
50            elliptic_curves: Vec::new(),
51            elliptic_curve_point_formats: Vec::new(),
52            signature_algorithms: Vec::new(),
53            sni: None,
54            alpn: None,
55            metadata: FingerprintMetadata::new(),
56        };
57        sig.id = sig.calculate_id();
58        sig
59    }
60
61    /// Calculates the fingerprint ID (based on signature trait)
62    pub fn calculate_id(&self) -> String {
63        let mut hasher = Sha256::new();
64        hasher.update(self.version.to_u16().to_be_bytes());
65        hasher.update(self.cipher_suites_without_grease().len().to_be_bytes());
66        for &cs in &self.cipher_suites_without_grease() {
67            hasher.update(cs.to_be_bytes());
68        }
69        hasher.update(self.extensions_without_grease().len().to_be_bytes());
70        for &ext in &self.extensions_without_grease() {
71            hasher.update(ext.to_be_bytes());
72        }
73        for &curve in &self.elliptic_curves {
74            hasher.update(curve.to_be_bytes());
75        }
76        if let Some(ref sni) = self.sni {
77            hasher.update(sni.as_bytes());
78        }
79        if let Some(ref alpn) = self.alpn {
80            hasher.update(alpn.as_bytes());
81        }
82        format!("{:x}", hasher.finalize())
83    }
84
85    /// Gets filtered cipher suites without GREASE values
86    pub fn cipher_suites_without_grease(&self) -> Vec<u16> {
87        filter_grease_values(&self.cipher_suites)
88    }
89
90    /// Gets filtered extensions without GREASE values
91    pub fn extensions_without_grease(&self) -> Vec<u16> {
92        filter_grease_values(&self.extensions)
93    }
94
95    /// Gets filtered signature algorithms without GREASE values
96    pub fn signature_algorithms_without_grease(&self) -> Vec<u16> {
97        filter_grease_values(&self.signature_algorithms)
98    }
99
100    /// Checks whether the signature includes GREASE values
101    pub fn has_grease(&self) -> bool {
102        self.cipher_suites.iter().any(|&v| is_grease_value(v))
103            || self.extensions.iter().any(|&v| is_grease_value(v))
104            || self
105                .signature_algorithms
106                .iter()
107                .any(|&v| is_grease_value(v))
108    }
109
110    /// Compares two signatures for similarity (ignoring GREASE values)
111    ///
112    /// # Parameters
113    /// * `other` - Another signature to compare
114    ///
115    /// # Returns
116    /// * `true` if signatures are similar (ignoring GREASE values), `false` otherwise
117    pub fn similar_to(&self, other: &Self) -> bool {
118        self.version == other.version
119            && self.cipher_suites_without_grease() == other.cipher_suites_without_grease()
120            && self.extensions_without_grease() == other.extensions_without_grease()
121            && self.signature_algorithms_without_grease()
122                == other.signature_algorithms_without_grease()
123            && self.elliptic_curves == other.elliptic_curves
124            && self.elliptic_curve_point_formats == other.elliptic_curve_point_formats
125            && self.sni == other.sni
126            && self.alpn == other.alpn
127    }
128
129    /// Calculates signature hash value (for fast comparison)
130    /// Uses filtered GREASE value
131    pub fn hash(&self) -> u64 {
132        let mut hasher = StableHashBuilder::new();
133        hasher.write_u16(self.version.to_u16());
134        hasher.write_u16_slice(&self.cipher_suites_without_grease());
135        hasher.write_u16_slice(&self.extensions_without_grease());
136        hasher.write_u16_slice(&self.signature_algorithms_without_grease());
137        hasher.write_u16_slice(&self.elliptic_curves);
138        hasher.write_u8_slice(&self.elliptic_curve_point_formats);
139        hasher.write_option_str(self.sni.as_deref());
140        hasher.write_option_str(self.alpn.as_deref());
141        hasher.finish()
142    }
143}

145impl Fingerprint for ClientHelloSignature {
146    fn fingerprint_type(&self) -> FingerprintType {
147        FingerprintType::Tls
148    }
149
150    fn id(&self) -> String {
151        self.id.clone()
152    }
153
154    fn metadata(&self) -> &FingerprintMetadata {
155        &self.metadata
156    }
157
158    fn metadata_mut(&mut self) -> &mut FingerprintMetadata {
159        &mut self.metadata
160    }
161
162    fn hash(&self) -> u64 {
163        self.hash()
164    }
165
166    fn similar_to(&self, other: &dyn Fingerprint) -> bool {
167        if other.fingerprint_type() != FingerprintType::Tls {
168            return false;
169        }
170
171        // Try to convert to ClientHelloSignature
172        // Due to trait limitations, we can only compare hash values
173        // Actual usage should convert through type conversion from comparison
174        self.hash() == other.hash()
175    }
176
177    fn to_string(&self) -> String {
178        format!(
179            "ClientHelloSignature(id={}, version={:?}, cipher_suites={}, extensions={})",
180            self.id,
181            self.version,
182            self.cipher_suites_without_grease().len(),
183            self.extensions_without_grease().len()
184        )
185    }
186}

194#[cfg(test)]
195mod tests {
196    use super::*;
197
198    #[test]
199    fn test_similar_to() {
200        let mut sig1 = ClientHelloSignature::new();
201        sig1.version = TlsVersion::V1_2;
202        sig1.cipher_suites = vec![0x0a0a, 0x0017, 0x1a1a]; // Including GREASE
203        sig1.extensions = vec![0x0000, 0x0010];
204
205        let mut sig2 = ClientHelloSignature::new();
206        sig2.version = TlsVersion::V1_2;
207        sig2.cipher_suites = vec![0x0017, 0x2a2a]; // Different GREASE, but filtered result should be same
208        sig2.extensions = vec![0x0000, 0x0010];
209
210        // Filtered GREASE should be the same
211        assert_eq!(
212            sig1.cipher_suites_without_grease(),
213            sig2.cipher_suites_without_grease()
214        );
215        assert!(sig1.similar_to(&sig2));
216    }
217
218    #[test]
219    fn test_has_grease() {
220        let mut sig = ClientHelloSignature::new();
221        assert!(!sig.has_grease());
222
223        sig.cipher_suites = vec![0x0a0a, 0x0017];
224        assert!(sig.has_grease());
225
226        sig.cipher_suites = vec![0x0017];
227        sig.extensions = vec![0x1a1a];
228        assert!(sig.has_grease());
229    }
