1/// Redis Integration for Rate Limiting
2///
3/// Provides connection pooling and distributed quota management via Redis.
4/// Used by the RateLimiter service for shared state across multiple gateway instances.
5use std::time::Duration;
6
7// Redis 配置常量
8const DEFAULT_REDIS_POOL_SIZE: u32 = 10; // 默认连接池大小
9const DEFAULT_REDIS_TIMEOUT_SECS: u64 = 5; // 默认超时（秒）
10const DEFAULT_REDIS_COMMAND_TIMEOUT_SECS: u64 = 2; // 默认命令超时（秒）
11
12/// Redis connection configuration
13#[derive(Debug, Clone)]
14pub struct RedisConfig {
15    /// Redis server connection URL (e.g., redis://localhost:6379)
16    pub url: String,
17    /// Connection pool size
18    pub pool_size: u32,
19    /// Connection timeout
20    pub timeout: Duration,
21    /// Command timeout
22    pub command_timeout: Duration,
23}
24
25impl RedisConfig {
26    /// Create new Redis configuration
27    pub fn new(url: String) -> Self {
28        Self {
29            url,
30            pool_size: DEFAULT_REDIS_POOL_SIZE,
31            timeout: Duration::from_secs(DEFAULT_REDIS_TIMEOUT_SECS),
32            command_timeout: Duration::from_secs(DEFAULT_REDIS_COMMAND_TIMEOUT_SECS),
33        }
34    }
35
36    /// Set pool size for connections
37    pub fn with_pool_size(mut self, size: u32) -> Self {
38        self.pool_size = size;
39        self
40    }
41
42    /// Set connection timeout
43    pub fn with_timeout(mut self, timeout: Duration) -> Self {
44        self.timeout = timeout;
45        self
46    }
47}

49/// Redis backend error types
50#[derive(Debug, Clone)]
51pub enum RedisBackendError {
52    /// Connection failed
53    ConnectionError(String),
54    /// Command execution failed
55    CommandError(String),
56    /// Serialization failed
57    SerializationError(String),
58    /// Timeout
59    TimeoutError(String),
60}
61
62impl std::fmt::Display for RedisBackendError {
63    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
64        match self {
65            RedisBackendError::ConnectionError(e) => write!(f, "Redis connection error: {}", e),
66            RedisBackendError::CommandError(e) => write!(f, "Redis command error: {}", e),
67            RedisBackendError::SerializationError(e) => write!(f, "Serialization error: {}", e),
68            RedisBackendError::TimeoutError(e) => write!(f, "Redis timeout: {}", e),
69        }
70    }
71}
72
73impl std::error::Error for RedisBackendError {}
74
75/// Result type for Redis backend operations
76pub type RedisResult<T> = Result<T, RedisBackendError>;

341/// Rate limit state stored in Redis
342#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
343pub struct RedisQuotaEntry {
344    pub user_id: String,
345    pub available_tokens: f64,
346    pub last_refill: u64,
347    pub month_requests: u64,
348    pub month_start: u64,
349    pub tier: String,
350}
351
352impl RedisQuotaEntry {
353    /// Create a new quota entry
354    pub fn new(
355        user_id: String,
356        available_tokens: f64,
357        last_refill: u64,
358        month_requests: u64,
359        month_start: u64,
360        tier: String,
361    ) -> Self {
362        Self {
363            user_id,
364            available_tokens,
365            last_refill,
366            month_requests,
367            month_start,
368            tier,
369        }
370    }
371
372    /// Convert from UserQuota
373    pub fn from_user_quota(user_id: String, quota: &super::UserQuota) -> Self {
374        Self {
375            user_id,
376            available_tokens: quota.available_tokens,
377            last_refill: quota.last_refill,
378            month_requests: quota.month_requests,
379            month_start: quota.month_start,
380            tier: format!("{:?}", quota.tier),
381        }
382    }
383}
