15/// Fingerprint result, including fingerprint, User-Agent and standard HTTP headers
16#[derive(Debug)]
17pub struct FingerprintResult {
18    /// Profile name/ID (e.g., "chrome_133")
19    pub profile_id: String,
20    /// Matching User-Agent
21    pub user_agent: String,
22    /// Standard HTTP request headers (including global language support)
23    pub headers: fingerprint_headers::headers::HTTPHeaders,
24    /// Browser type
25    pub browser_type: BrowserType,
26}
27
28/// Browser type not found error
29#[derive(Debug, Clone)]
30pub struct BrowserNotFoundError {
31    pub browser: String,
32}
33
34impl std::fmt::Display for BrowserNotFoundError {
35    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
36        write!(f, "browser type not found: {}", self.browser)
37    }
38}
39
40impl std::error::Error for BrowserNotFoundError {}
41
42/// Randomly get a fingerprint and matching User-Agent
43/// Operating system will be randomly selected
44pub fn get_random_fingerprint() -> Result<FingerprintResult, String> {
45    get_random_fingerprint_with_os(None)
46}

48/// Randomly get a fingerprint and matching User-Agent, with specified operating system
49/// If os is None, then operating system is randomly selected
50///
51/// # Note
52/// When os is specified (Some), mobile profiles (e.g., Android, iOS) are automatically filtered out
53/// because mobile profiles have fixed User-Agent strings that cannot be changed to other operating systems
54pub fn get_random_fingerprint_with_os(
55    os: Option<OperatingSystem>,
56) -> Result<FingerprintResult, String> {
57    let clients = mapped_tls_clients();
58    if clients.is_empty() {
59        return Err("no TLS client profiles available".to_string());
60    }
61
62    // Get all available fingerprint names
63    // If OS is specified, filter out mobile profiles (mobile profiles have fixed UA that cannot change OS)
64    let names: Vec<String> = match os {
65        Some(_) => clients
66            .keys()
67            .filter(|name| !is_mobile_profile(name))
68            .cloned()
69            .collect(),
70        None => clients.keys().cloned().collect(),
71    };
72
73    // Randomly select one (thread-safe)
74    let name_refs: Vec<&str> = names.iter().map(|s| s.as_str()).collect();
75    let random_name = random_choice_string(&name_refs)
76        .ok_or_else(|| "failed to select random profile".to_string())?;
77
78    let profile = clients
79        .get(&random_name)
80        .ok_or_else(|| format!("profile {} not found", random_name))?;
81
82    let profile_id = profile.id();
83
84    // Get matching User-Agent
85    let ua = match os {
86        Some(os) => get_user_agent_by_profile_name_with_os(&random_name, os)?,
87        None => get_user_agent_by_profile_name(&random_name)?,
88    };
89
90    // Generate standard HTTP headers
91    let (browser_type_str, _) = infer_browser_from_profile_name(&random_name);
92    let is_mobile = is_mobile_profile(&random_name);
93    let browser_type = BrowserType::from_str(&browser_type_str).unwrap_or(BrowserType::Chrome);
94    let headers = generate_headers(browser_type, &ua, is_mobile);
95
96    Ok(FingerprintResult {
97        profile_id,
98        user_agent: ua,
99        headers,
100        browser_type,
101    })
102}

104/// Based on browser type randomly get fingerprint and User-Agent

106pub fn get_random_fingerprint_by_browser(

112/// Based on browser type randomly get fingerprint and User-Agent, 并specified operating system
113pub fn get_random_fingerprint_by_browser_with_os(
114    browser_type: &str,
115    os: Option<OperatingSystem>,
116) -> Result<FingerprintResult, Box<dyn std::error::Error>> {
117    if browser_type.is_empty() {
118        return Err("browser type cannot be empty".into());
119    }
120
121    let clients = mapped_tls_clients();
122    if clients.is_empty() {
123        return Err("no TLS client profiles available".into());
124    }
125
126    let browser_type_lower = browser_type.to_lowercase();
127
128    // filter out specified browser type fingerprint
129    let candidates: Vec<String> = clients
130        .keys()
131        .filter(|name| {
132            let name_lower = name.to_lowercase();
133            name_lower.starts_with(&format!("{}_", browser_type_lower))
134        })
135        .cloned()
136        .collect();
137
138    if candidates.is_empty() {
139        return Err(Box::new(BrowserNotFoundError {
140            browser: browser_type.to_string(),
141        }));
142    }
143
144    // randomly select an (thread security)
145    let candidate_refs: Vec<&str> = candidates.iter().map(|s| s.as_str()).collect();
146    let random_name = random_choice_string(&candidate_refs)
147        .ok_or_else(|| "failed to select random profile".to_string())?;
148
149    let profile = clients
150        .get(&random_name)
151        .ok_or_else(|| format!("profile {} not found", random_name))?;
152
153    let profile_id = profile.id();
154
155    // Get pair should User-Agent
156    let ua = match os {
157        Some(os) => get_user_agent_by_profile_name_with_os(&random_name, os)?,
158        None => get_user_agent_by_profile_name(&random_name)?,
159    };
160
161    // Generate standard HTTP Headers
162    let (browser_type_str, _) = infer_browser_from_profile_name(&random_name);
163    let is_mobile = is_mobile_profile(&random_name);
164    let browser_type_enum = BrowserType::from_str(&browser_type_str).unwrap_or(BrowserType::Chrome);
165    let headers = generate_headers(browser_type_enum, &ua, is_mobile);
166
167    Ok(FingerprintResult {
168        profile_id,
169        user_agent: ua,
170        headers,
171        browser_type: browser_type_enum,
172    })
173}
