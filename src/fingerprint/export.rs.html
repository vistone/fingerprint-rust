1//! configurationexportmodule
2//!
3//! will ClientHelloSpec export as JSON format, so thatprovideotherlanguage (如 Go uTLS)use.

5#[cfg(feature = "export")]
6use fingerprint_tls::tls_config::ClientHelloSpec;
7#[cfg(feature = "export")]
8use fingerprint_tls::tls_extensions::*;
9#[cfg(feature = "export")]
10use serde::{Deserialize, Serialize};
11
12/// exportconfigurationstruct
13#[cfg(feature = "export")]
14#[derive(Serialize, Deserialize, Debug)]
15pub struct ExportConfig {
16    pub cipher_suites: Vec<u16>,
17    pub compression_methods: Vec<u8>,
18    pub extensions: Vec<ExportExtension>,
19    pub tls_vers_min: u16,
20    pub tls_vers_max: u16,
21}
22
23/// export KeyShare
24#[cfg(feature = "export")]
25#[derive(Serialize, Deserialize, Debug)]
26pub struct ExportKeyShare {
27    pub group: u16,
28    pub data_hex: String,
29}
30
31/// export's extensionsenum
32#[cfg(feature = "export")]
33#[derive(Serialize, Deserialize, Debug)]
34#[serde(tag = "type", content = "data")]
35pub enum ExportExtension {
36    SNI, // onlytype，nocountdata (由clientrun when decide)
37    StatusRequest,
38    SupportedCurves(Vec<u16>),
39    SupportedPoints(Vec<u8>),
40    SignatureAlgorithms(Vec<u16>),
41    ALPN(Vec<String>),
42    ExtendedMasterSecret,
43    SessionTicket,
44    SupportedVersions(Vec<u16>),
45    PSKKeyExchangeModes(Vec<u8>),
46    KeyShare(Vec<ExportKeyShare>),
47    SCT,
48    RenegotiationInfo(u8),
49    ApplicationSettings(Vec<String>),
50    CompressCertificate(Vec<u16>),
51    PreSharedKey,
52    GREASE(u16),
53    Padding { padding_len: usize, will_pad: bool },
54    ECH(u16),
55    Unknown(u16),
56}
57
58/// will ClientHelloSpec convert to JSON string
59#[cfg(feature = "export")]
60pub fn export_config_json(spec: &ClientHelloSpec) -> Result<String, serde_json::Error> {
61    let export = ExportConfig::from(spec);
62    serde_json::to_string_pretty(&export)
63}
64
65#[cfg(feature = "export")]
66impl From<&ClientHelloSpec> for ExportConfig {
67    fn from(spec: &ClientHelloSpec) -> Self {
68        Self {
69            cipher_suites: spec.cipher_suites.clone(),
70            compression_methods: spec.compression_methods.clone(),
71            tls_vers_min: spec.tls_vers_min,
72            tls_vers_max: spec.tls_vers_max,
73            extensions: spec
74                .extensions
75                .iter()
76                .map(|ext| {
77                    // use as_any performtowarddowntransform
78                    let any_ext = ext.as_any();
79
80                    if let Some(e) = any_ext.downcast_ref::<UtlsGREASEExtension>() {
81                        return ExportExtension::GREASE(e.value);
82                    }
83
84                    if let Some(_e) = any_ext.downcast_ref::<SNIExtension>() {
85                        return ExportExtension::SNI;
86                    }
87
88                    if let Some(_e) = any_ext.downcast_ref::<StatusRequestExtension>() {
89                        return ExportExtension::StatusRequest;
90                    }
91
92                    if let Some(e) = any_ext.downcast_ref::<SupportedCurvesExtension>() {
93                        return ExportExtension::SupportedCurves(e.curves.clone());
94                    }
95
96                    if let Some(e) = any_ext.downcast_ref::<SupportedPointsExtension>() {
97                        return ExportExtension::SupportedPoints(e.supported_points.clone());
98                    }
99
100                    if let Some(e) = any_ext.downcast_ref::<SignatureAlgorithmsExtension>() {
101                        // SignatureScheme is u16
102                        return ExportExtension::SignatureAlgorithms(
103                            e.supported_signature_algorithms.clone(),
104                        );
105                    }
106
107                    if let Some(e) = any_ext.downcast_ref::<ALPNExtension>() {
108                        return ExportExtension::ALPN(e.alpn_protocols.clone());
109                    }
110
111                    if let Some(_e) = any_ext.downcast_ref::<ExtendedMasterSecretExtension>() {
112                        return ExportExtension::ExtendedMasterSecret;
113                    }
114
115                    if let Some(_e) = any_ext.downcast_ref::<SessionTicketExtension>() {
116                        return ExportExtension::SessionTicket;
117                    }
118
119                    if let Some(e) = any_ext.downcast_ref::<SupportedVersionsExtension>() {
120                        return ExportExtension::SupportedVersions(e.versions.clone());
121                    }
122
123                    if let Some(e) = any_ext.downcast_ref::<PSKKeyExchangeModesExtension>() {
124                        return ExportExtension::PSKKeyExchangeModes(e.modes.clone());
125                    }
126
127                    if let Some(e) = any_ext.downcast_ref::<KeyShareExtension>() {
128                        let shares = e
129                            .key_shares
130                            .iter()
131                            .map(|ks| ExportKeyShare {
132                                group: ks.group,
133                                data_hex: {
134                                    #[cfg(feature = "hex")]
135                                    {
136                                        hex::encode(&ks.data)
137                                    }
138                                    #[cfg(not(feature = "hex"))]
139                                    {
140                                        // Ifno hex feature, usehexadecimalformatmanualencoding
141                                        ks.data
142                                            .iter()
143                                            .map(|b| format!("{:02x}", b))
144                                            .collect::<String>()
145                                    }
146                                },
147                            })
148                            .collect();
149                        return ExportExtension::KeyShare(shares);
150                    }
151
152                    if let Some(_e) = any_ext.downcast_ref::<SCTExtension>() {
153                        return ExportExtension::SCT;
154                    }
155
156                    if let Some(e) = any_ext.downcast_ref::<RenegotiationInfoExtension>() {
157                        return ExportExtension::RenegotiationInfo(e.renegotiation);
158                    }
159
160                    if let Some(e) = any_ext.downcast_ref::<ApplicationSettingsExtensionNew>() {
161                        return ExportExtension::ApplicationSettings(e.supported_protocols.clone());
162                    }
163
164                    if let Some(e) = any_ext.downcast_ref::<UtlsCompressCertExtension>() {
165                        return ExportExtension::CompressCertificate(e.algorithms.clone());
166                    }
167
168                    if let Some(_e) = any_ext.downcast_ref::<UtlsPreSharedKeyExtension>() {
169                        return ExportExtension::PreSharedKey;
170                    }
171
172                    if let Some(e) = any_ext.downcast_ref::<GREASEEncryptedClientHelloExtension>() {
173                        return ExportExtension::ECH(e.value);
174                    }
175
176                    if let Some(e) = any_ext.downcast_ref::<UtlsPaddingExtension>() {
177                        return ExportExtension::Padding {
178                            padding_len: e.padding_len,
179                            will_pad: e.will_pad,
180                        };
181                    }
182
183                    ExportExtension::Unknown(ext.extension_id())
184                })
185                .collect(),
186        }
187    }
188}
