66/// Advanced anomaly detector using multiple ML techniques
67pub struct AdvancedAnomalyDetector {
68    // Simple implementation for now - baseline for normal behavior
69    baseline_normal: Vec<f32>,
70}
71
72impl AdvancedAnomalyDetector {
73    /// Create new anomaly detector with default parameters
74    pub fn new() -> Self {
75        Self {
76            baseline_normal: vec![0.1, 0.15, 0.12, 0.18, 0.14],
77        }
78    }
79
80    /// Detect anomalies in the given fingerprint
81    pub fn detect_anomalies(&self, fingerprint: &FingerprintVector) -> AnomalyDetectionResult {
82        // Simple distance-based detection as placeholder for more sophisticated ML algorithms
83        let distance: f32 = self
84            .baseline_normal
85            .iter()
86            .zip(fingerprint.features.iter())
87            .map(|(a, b)| (a - b).powi(2))
88            .sum::<f32>()
89            .sqrt();
90
91        let anomaly_score = (distance / self.baseline_normal.len() as f32).min(1.0);
92
93        let classification = if anomaly_score < 0.1 {
94            AnomalyClassification::Normal
95        } else if anomaly_score < 0.2 {
96            AnomalyClassification::Suspicious
97        } else if anomaly_score < 0.3 {
98            AnomalyClassification::Anomalous
99        } else {
100            AnomalyClassification::Critical
101        };
102
103        AnomalyDetectionResult {
104            anomaly_score,
105            confidence: fingerprint.confidence,
106            classification,
107            explanation: format!("Distance from baseline: {:.3}", distance),
108        }
109    }
110}

112/// ML fingerprint matcher
113pub struct FingerprintMatcher {
114    profiles: HashMap<String, FingerprintVector>,
115}
116
117impl FingerprintMatcher {
118    /// Create new matcher
119    pub fn new() -> Self {
120        FingerprintMatcher {
121            profiles: HashMap::new(),
122        }
123    }
124
125    /// Add reference fingerprint
126    pub fn add_reference(&mut self, id: String, vector: FingerprintVector) {
127        self.profiles.insert(id, vector);
128    }
129
130    /// Find most similar fingerprint
131    pub fn find_most_similar(&self, query: &FingerprintVector) -> Option<(String, f32)> {
132        self.profiles
133            .iter()
134            .map(|(id, profile)| {
135                let similarity = self.cosine_similarity(query, profile);
136                (id.clone(), similarity)
137            })
138            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
139    }
140
141    /// Calculate cosine similarity
142    fn cosine_similarity(&self, a: &FingerprintVector, b: &FingerprintVector) -> f32 {
143        let dot_product: f32 = a.features.iter().zip(&b.features).map(|(x, y)| x * y).sum();
144        let magnitude_a: f32 = a.features.iter().map(|x| x * x).sum::<f32>().sqrt();
145        let magnitude_b: f32 = b.features.iter().map(|x| x * x).sum::<f32>().sqrt();
146
147        if magnitude_a == 0.0 || magnitude_b == 0.0 {
148            0.0
149        } else {
150            dot_product / (magnitude_a * magnitude_b)
151        }
152    }
153
154    /// Get all matches
155    pub fn get_matches(&self, query: &FingerprintVector, threshold: f32) -> Vec<(String, f32)> {
156        self.profiles
157            .iter()
158            .filter_map(|(id, profile)| {
159                let similarity = self.cosine_similarity(query, profile);
160                if similarity >= threshold {
161                    Some((id.clone(), similarity))
162                } else {
163                    None
164                }
165            })
166            .collect()
167    }
168}

170/// Behavior classifier
171pub struct BehaviorClassifier {
172    normal_patterns: Vec<FingerprintVector>,
173    suspicious_patterns: Vec<FingerprintVector>,
174}
175
176impl BehaviorClassifier {
177    /// Classify behavior
178    pub fn classify(&self, fingerprint: &FingerprintVector) -> BehaviorClass {
179        // Simple classification based on distance to known patterns
180        let normal_distances: Vec<f32> = self
181            .normal_patterns
182            .iter()
183            .map(|pattern| self.euclidean_distance(fingerprint, pattern))
184            .collect();
185
186        let suspicious_distances: Vec<f32> = self
187            .suspicious_patterns
188            .iter()
189            .map(|pattern| self.euclidean_distance(fingerprint, pattern))
190            .collect();
191
192        let avg_normal = normal_distances.iter().sum::<f32>() / normal_distances.len() as f32;
193        let avg_suspicious =
194            suspicious_distances.iter().sum::<f32>() / suspicious_distances.len() as f32;
195
196        if avg_normal < avg_suspicious * 0.8 {
197            BehaviorClass::Human
198        } else if avg_suspicious < avg_normal * 0.8 {
199            BehaviorClass::Bot
200        } else if avg_normal < 1.0 && avg_suspicious < 1.0 {
201            BehaviorClass::Normal
202        } else {
203            BehaviorClass::Suspicious
204        }
205    }
206
207    /// Calculate risk score
208    pub fn calculate_risk_score(&self, fingerprint: &FingerprintVector) -> f32 {
209        let normal_distances: Vec<f32> = self
210            .normal_patterns
211            .iter()
212            .map(|pattern| self.euclidean_distance(fingerprint, pattern))
213            .collect();
214
215        let suspicious_distances: Vec<f32> = self
216            .suspicious_patterns
217            .iter()
218            .map(|pattern| self.euclidean_distance(fingerprint, pattern))
219            .collect();
220
221        let avg_normal = if normal_distances.is_empty() {
222            1.0
223        } else {
224            normal_distances.iter().sum::<f32>() / normal_distances.len() as f32
225        };
226
227        let avg_suspicious = if suspicious_distances.is_empty() {
228            0.0
229        } else {
230            suspicious_distances.iter().sum::<f32>() / suspicious_distances.len() as f32
231        };
232
233        // Risk score: higher when closer to suspicious patterns
234        (avg_suspicious / (avg_normal + avg_suspicious)).clamp(0.0, 1.0)
235    }
236
237    /// Calculate variance
238    fn euclidean_distance(&self, a: &FingerprintVector, b: &FingerprintVector) -> f32 {
239        a.features
240            .iter()
241            .zip(&b.features)
242            .map(|(x, y)| (x - y).powi(2))
243            .sum::<f32>()
244            .sqrt()
245    }
246}
