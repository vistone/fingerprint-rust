15/// fontfingerprint
16#[derive(Debug, Clone)]
17pub struct FontFingerprint {
18    /// detect到ofsystemfontlist
19    pub system_fonts: Vec<String>,
20    /// fontloadtime (ms)
21    pub loading_times: Vec<u64>,
22    /// uniquefontfingerprinthash
23    pub unique_hash: String,
24    /// fontcount
25    pub font_count: usize,
26    /// supportofsubset
27    pub supported_subsets: Vec<String>,
28    /// renderingfeatures
29    pub rendering_features: Vec<String>,
30}

32/// fonterrortype
33#[derive(Debug)]
34pub enum FontError {
35    /// invaliddata
36    InvalidData,
37    /// enumerationfailure
38    EnumerationFailed(String),
39    /// othererror
40    Other(String),
41}
42
43impl std::fmt::Display for FontError {
44    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
45        match self {
46            FontError::InvalidData => write!(f, "Invalid font data"),
47            FontError::EnumerationFailed(msg) => write!(f, "Enumeration failed: {}", msg),
48            FontError::Other(msg) => write!(f, "Error: {}", msg),
49        }
50    }
51}
52
53impl std::error::Error for FontError {}

55/// fontanalyzer
56pub struct FontAnalyzer;
57
58impl FontAnalyzer {
59    /// analyzesystemfont
60    pub fn analyze(system_fonts: &[&str]) -> Result<FontFingerprint, FontError> {
61        if system_fonts.is_empty() {
62            return Err(FontError::InvalidData);
63        }
64
65        // converttostringvector
66        let fonts: Vec<String> = system_fonts.iter().map(|s| s.to_string()).collect();
67
68        // calculateloadtime
69        let loading_times = Self::calculate_loading_times(&fonts);
70
71        // generateuniquehash
72        let unique_hash = Self::generate_font_hash(&fonts);
73
74        // detectsubsetsupport
75        let supported_subsets = Self::detect_subsets(&fonts);
76
77        // getrenderingfeatures
78        let rendering_features = Self::get_rendering_features(&fonts);
79
80        Ok(FontFingerprint {
81            system_fonts: fonts.clone(),
82            loading_times,
83            unique_hash,
84            font_count: fonts.len(),
85            supported_subsets,
86            rendering_features,
87        })
88    }
89
90    /// calculatefontloadtime
91    fn calculate_loading_times(fonts: &[String]) -> Vec<u64> {
92        // based onfontnamelengthandfeaturesofsimulatedtime
93        fonts
94            .iter()
95            .map(|f| {
96                let base = f.len() as u64 * 10;
97                let hash = f.chars().map(|c| c as u64).sum::<u64>();
98                (base + hash % 50)
99            })
100            .collect()
101    }
102
103    /// generatefonthash
104    fn generate_font_hash(fonts: &[String]) -> String {
105        let hash_input = fonts.join(":");
106        let hash_value = hash_input
107            .chars()
108            .fold(0u64, |acc, c| acc.wrapping_mul(31).wrapping_add(c as u64));
109        format!("{:x}", hash_value)
110    }
111
112    /// detectsupportofsubset
113    fn detect_subsets(fonts: &[String]) -> Vec<String> {
114        let mut subsets = HashSet::new();
115
116        // based onfontnamedetectsubset
117        for font in fonts {
118            let lower = font.to_lowercase();
119            if lower.contains("cjk") {
120                subsets.insert("cjk".to_string());
121            }
122            if lower.contains("arabic") {
123                subsets.insert("arabic".to_string());
124            }
125            if lower.contains("hebrew") {
126                subsets.insert("hebrew".to_string());
127            }
128            if lower.contains("thai") {
129                subsets.insert("thai".to_string());
130            }
131        }
132
133        // defaultsubset
134        subsets.insert("latin".to_string());
135
136        subsets.into_iter().collect()
137    }
138
139    /// getrenderingfeatures
140    fn get_rendering_features(_fonts: &[String]) -> Vec<String> {
141        vec![
142            "anti-aliasing".to_string(),
143            "hinting".to_string(),
144            "kerning".to_string(),
145            "ligatures".to_string(),
146        ]
147    }
148}

150/// fontsystemdetector
151pub struct FontSystemDetector;
152
153impl FontSystemDetector {
154    /// detectoperating systemfont
155    pub fn detect_system() -> FontFingerprint {
156        let default_fonts = vec![
157            "Arial",
158            "Times New Roman",
159            "Courier New",
160            "Verdana",
161            "Georgia",
162            "Trebuchet MS",
163            "Comic Sans MS",
164            "Impact",
165        ];
166
167        FontAnalyzer::analyze(&default_fonts).unwrap_or_else(|_| FontFingerprint {
168            system_fonts: Vec::new(),
169            loading_times: Vec::new(),
170            unique_hash: "unknown".to_string(),
171            font_count: 0,
172            supported_subsets: vec!["latin".to_string()],
173            rendering_features: vec!["anti-aliasing".to_string()],
174        })
175    }
176}
