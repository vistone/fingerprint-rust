3//! # Hardware Fingerprint Module
4//!
5//! 硬件指纹识别模块

7//! 提供 GPU、CPU、内存等硬件识别功能

9/// 硬件fingerprint
10#[derive(Debug, Clone)]
11pub struct HardwareFingerprint {
12    /// CPU 型号
13    pub cpu_model: String,
14    /// CPU 核心数
15    pub cpu_cores: u32,
16    /// GPU 型号
17    pub gpu_model: String,
18    /// GPU memory
19    pub gpu_memory_gb: u32,
20    /// systemmemory (GB)
21    pub system_memory_gb: u64,
22    /// 屏幕 DPI
23    pub screen_dpi: f32,
24    /// 屏幕resolution
25    pub screen_resolution: (u32, u32),
26    /// 设备type
27    pub device_type: DeviceType,
28}
29
30/// 设备type
31#[derive(Debug, Clone, Copy, PartialEq, Eq)]
32pub enum DeviceType {
33    /// 桌面
34    Desktop,
35    /// 笔记本
36    Laptop,
37    /// 平板
38    Tablet,
39    /// 手机
40    Phone,
41    /// unknown
42    Unknown,
43}
44
45/// 硬件errortype
46#[derive(Debug)]
47pub enum HardwareError {
48    /// invaliddata
49    InvalidData,
50    /// detectfailure
51    DetectionFailed(String),
52    /// othererror
53    Other(String),
54}
55
56impl std::fmt::Display for HardwareError {
57    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
58        match self {
59            HardwareError::InvalidData => write!(f, "Invalid hardware data"),
60            HardwareError::DetectionFailed(msg) => write!(f, "Detection failed: {}", msg),
61            HardwareError::Other(msg) => write!(f, "Error: {}", msg),
62        }
63    }
64}
65
66impl std::error::Error for HardwareError {}

68/// 硬件detector
69pub struct HardwareDetector;
70
71impl HardwareDetector {
72    /// detect硬件info
73    pub fn detect(
74        cpu_cores: u32,
75        gpu_model: &str,
76        system_memory_gb: u64,
77        screen_dpi: f32,
78        screen_width: u32,
79        screen_height: u32,
80    ) -> Result<HardwareFingerprint, HardwareError> {
81        if system_memory_gb == 0 || cpu_cores == 0 {
82            return Err(HardwareError::InvalidData);
83        }
84
85        let cpu_model = Self::identify_cpu(cpu_cores);
86        let device_type = Self::identify_device_type(screen_width, screen_height, system_memory_gb);
87        let gpu_memory = Self::estimate_gpu_memory(gpu_model);
88
89        Ok(HardwareFingerprint {
90            cpu_model,
91            cpu_cores,
92            gpu_model: gpu_model.to_string(),
93            gpu_memory_gb: gpu_memory,
94            system_memory_gb,
95            screen_dpi,
96            screen_resolution: (screen_width, screen_height),
97            device_type,
98        })
99    }
100
101    /// recognition CPU 型号
102    fn identify_cpu(cores: u32) -> String {
103        match cores {
104            1..=2 => "Intel Core i5".to_string(),
105            3..=4 => "Intel Core i7".to_string(),
106            5..=8 => "Intel Core i9".to_string(),
107            _ => format!("CPU ({} cores)", cores),
108        }
109    }
110
111    /// recognition设备type
112    fn identify_device_type(width: u32, height: u32, memory: u64) -> DeviceType {
113        match (width, height, memory) {
114            // 手机条件要在Tablet之前，因to更具体
115            (w, h, _) if (w <= 480 && h <= 960) || (w <= 540 && h <= 960) => DeviceType::Phone,
116            // Tablet: 7-13 英寸屏幕通常to 600-1024 宽
117            (w, h, _) if (w <= 800 && h <= 1280) || (w <= 1280 && h <= 800) => DeviceType::Tablet,
118            // 桌面: memory多于16GB
119            (_, _, m) if m > 16 => DeviceType::Desktop,
120            // defaultto笔记本
121            _ => DeviceType::Laptop,
122        }
123    }
124
125    /// 估计 GPU memory
126    fn estimate_gpu_memory(gpu_model: &str) -> u32 {
127        match gpu_model {
128            gpu if gpu.contains("RTX 4090") => 24,
129            gpu if gpu.contains("RTX 4080") => 16,
130            gpu if gpu.contains("RTX 4070") => 12,
131            gpu if gpu.contains("M1") || gpu.contains("M2") => 8,
132            gpu if gpu.contains("Intel") => 2,
133            _ => 4,
134        }
135    }
136}

138/// 硬件fingerprint匹配器
139pub struct HardwareProfileMatcher;
140
141impl HardwareProfileMatcher {
142    /// 匹配硬件configure
143    pub fn match_profile(hardware: &HardwareFingerprint) -> Option<String> {
144        match (hardware.cpu_cores, hardware.device_type) {
145            (4, DeviceType::Laptop) => Some("MacBook Pro".to_string()),
146            (8, DeviceType::Desktop) => Some("Gaming PC".to_string()),
147            (2, DeviceType::Tablet) => Some("iPad".to_string()),
148            _ => None,
149        }
150    }
151
152    /// calculate硬件similarity
153    pub fn calculate_similarity(hw1: &HardwareFingerprint, hw2: &HardwareFingerprint) -> f32 {
154        let mut score = 0.0;
155        if hw1.cpu_cores == hw2.cpu_cores {
156            score += 0.2;
157        }
158        if hw1.gpu_model == hw2.gpu_model {
159            score += 0.2;
160        }
161        if hw1.system_memory_gb == hw2.system_memory_gb {
162            score += 0.2;
163        }
164        if hw1.device_type == hw2.device_type {
165            score += 0.2;
166        }
167        if (hw1.screen_dpi - hw2.screen_dpi).abs() < 10.0 {
168            score += 0.2;
169        }
170        score
171    }
172}
