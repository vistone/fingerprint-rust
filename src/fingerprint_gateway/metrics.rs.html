1//! Prometheus metrics collection
2//!
3//! This module provides comprehensive metrics for monitoring the API Gateway:
4//! - HTTP request counters
5//! - Rate limit statistics  
6//! - Response time histograms
7//! - Redis connection health
8
9use prometheus::{
10    opts, register_histogram_vec, register_int_counter_vec, register_int_gauge, Encoder,
11    HistogramVec, IntCounterVec, IntGauge, TextEncoder,
12};
13use std::time::Instant;

15lazy_static::lazy_static! {
16    /// Total HTTP requests by method, endpoint, and status code
17    pub static ref HTTP_REQUESTS_TOTAL: IntCounterVec = register_int_counter_vec!(
18        opts!("fingerprint_gateway_http_requests_total", "Total HTTP requests"),
19        &["method", "endpoint", "status"]
20    ).unwrap();
21
22    /// Rate limit check results by tier and result (allowed/blocked)
23    pub static ref RATE_LIMIT_CHECKS_TOTAL: IntCounterVec = register_int_counter_vec!(
24        opts!("fingerprint_gateway_rate_limit_checks_total", "Total rate limit checks"),
25        &["tier", "result"]
26    ).unwrap();
27
28    /// Rate limit quota usage by tier
29    pub static ref RATE_LIMIT_QUOTA_USED: IntCounterVec = register_int_counter_vec!(
30        opts!("fingerprint_gateway_rate_limit_quota_used", "Rate limit quota used"),
31        &["tier", "period"]
32    ).unwrap();
33
34    /// HTTP request duration histogram by method and endpoint
35    pub static ref HTTP_REQUEST_DURATION_SECONDS: HistogramVec = register_histogram_vec!(
36        "fingerprint_gateway_http_request_duration_seconds",
37        "HTTP request duration in seconds",
38        &["method", "endpoint"],
39        vec![0.001, 0.005, 0.010, 0.025, 0.050, 0.100, 0.250, 0.500, 1.0, 2.5, 5.0, 10.0]

42    /// Redis connection pool active connections

44        opts!("fingerprint_gateway_redis_connections_active", "Active Redis connections")
45    ).unwrap();
46
47    /// Redis operations counter
48    pub static ref REDIS_OPERATIONS_TOTAL: IntCounterVec = register_int_counter_vec!(

51    ).unwrap();

54/// Request timer for tracking request duration
55pub struct RequestTimer {
56    start: Instant,
57    method: String,
58    endpoint: String,
59}
60
61impl RequestTimer {
62    /// Start a new request timer
63    pub fn new(method: String, endpoint: String) -> Self {
64        Self {
65            start: Instant::now(),
66            method,
67            endpoint,
68        }
69    }
70
71    /// Record the request duration
72    pub fn observe(self) {
73        let duration = self.start.elapsed().as_secs_f64();
74        HTTP_REQUEST_DURATION_SECONDS
75            .with_label_values(&[&self.method, &self.endpoint])
76            .observe(duration);
77    }
78}
79
80/// Record an HTTP request
81pub fn record_http_request(method: &str, endpoint: &str, status: u16) {
82    HTTP_REQUESTS_TOTAL
83        .with_label_values(&[method, endpoint, &status.to_string()])
84        .inc();
85}
86
87/// Record a rate limit check
88pub fn record_rate_limit_check(tier: &str, allowed: bool) {
89    let result = if allowed { "allowed" } else { "blocked" };
90    RATE_LIMIT_CHECKS_TOTAL
91        .with_label_values(&[tier, result])
92        .inc();
93}
94
95/// Record quota usage
96pub fn record_quota_usage(tier: &str, period: &str) {
97    RATE_LIMIT_QUOTA_USED
98        .with_label_values(&[tier, period])
99        .inc();
100}
101
102/// Record Redis operation
103pub fn record_redis_operation(operation: &str, success: bool) {
104    let status = if success { "success" } else { "error" };
105    REDIS_OPERATIONS_TOTAL
106        .with_label_values(&[operation, status])
107        .inc();
108}
109
110/// Update Redis connection count
111pub fn update_redis_connections(count: i64) {
112    REDIS_CONNECTIONS_ACTIVE.set(count);
113}
114
115/// Gather all metrics in Prometheus text format
116pub fn gather_metrics() -> Result<String, Box<dyn std::error::Error>> {
117    let encoder = TextEncoder::new();
118    let metric_families = prometheus::gather();
119    let mut buffer = vec![];
120    encoder.encode(&metric_families, &mut buffer)?;
121    Ok(String::from_utf8(buffer)?)
122}

148    #[test]
149    fn test_gather_metrics() {
150        // Record some metrics first
151        record_http_request("GET", "/health", 200);

163                || metrics.contains("fingerprint_gateway_redis_connections_active")
164                || !metrics.is_empty(),
165            "Metrics output should contain gateway metric names or be non-empty"
