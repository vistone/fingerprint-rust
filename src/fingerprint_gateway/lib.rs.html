1//! # fingerprint-gateway
2//!
3//! High-performance API Gateway with rate limiting for fingerprint-rust.
4//!
5//! ## Features
6//!
7//! - **Rate Limiting**: Token bucket algorithm with Redis backend
8//! - **Quota Management**: Multi-tier quota system (Free, Pro, Enterprise, Partner)
9//! - **Metrics**: Prometheus metrics for monitoring
10//! - **High Performance**: Built on actix-web, 10x faster than Python FastAPI
11//! - **Type Safe**: Full Rust type safety
12//!
13//! ## Architecture
14//!
15//! ```text
16//! ┌─────────────────┐
17//! │   HTTP Client   │
18//! └────────┬────────┘
19//!          │
20//!          ▼
21//! ┌─────────────────┐
22//! │  actix-web API  │
23//! │   Gateway       │
24//! └────────┬────────┘
25//!          │
26//!          ├──────────────┐
27//!          │              │
28//!          ▼              ▼
29//! ┌─────────────┐  ┌─────────────┐
30//! │ Rate Limiter│  │  Prometheus │
31//! │  (Redis)    │  │   Metrics   │
32//! └─────────────┘  └─────────────┘
33//! ```
34//!
35//! ## Usage
36//!
37//! ### As a library
38//!
39//! ```rust,no_run
40//! use fingerprint_gateway::{GatewayConfig, run_server};
41//!
42//! #[tokio::main]
43//! async fn main() -> Result<(), Box<dyn std::error::Error>> {
44//!     let config = GatewayConfig::from_env()?;
45//!     run_server(config).await.map_err(|e| e.into())
46//! }
47//! ```
48//!
49//! ### As a binary
50//!
51//! ```bash
52//! cargo run --bin gateway --release
53//! ```

76/// Run the API Gateway server
77///
78/// # Arguments
79///
80/// * `config` - Gateway configuration
81///
82/// # Returns
83///
84/// Returns `Ok(())` when server shuts down gracefully, or an error if startup fails
85///
86/// # Example
87///
88/// ```rust,no_run
89/// use fingerprint_gateway::{GatewayConfig, run_server};
90///
91/// #[tokio::main]
92/// async fn main() -> Result<(), Box<dyn std::error::Error>> {
93///     let config = GatewayConfig::default();
94///     run_server(config).await.map_err(|e| e.into())
95/// }
96/// ```
97pub async fn run_server(config: GatewayConfig) -> Result<()> {
98    // Initialize tracing
99    tracing_subscriber::fmt()
100        .with_env_filter(
101            tracing_subscriber::EnvFilter::try_from_default_env()
102                .unwrap_or_else(|_| tracing_subscriber::EnvFilter::new("info")),
103        )
104        .json()
105        .init();
106
107    info!(
108        "Starting fingerprint-gateway v{}",
109        env!("CARGO_PKG_VERSION")
110    );
111    info!("Configuration: {:?}", config);
112
113    // Initialize rate limiter
114    let rate_limiter = Arc::new(
115        RateLimiter::new(config.redis_url.clone())
116            .await
117            .map_err(|e| {
118                warn!("Failed to initialize rate limiter: {}", e);
119                e
120            })?,
121    );
122
123    info!("Rate limiter initialized with Redis backend");
124
125    // Initialize API key validator
126    let api_key_validator = Arc::new(auth::ApiKeyValidator::new());
127    info!("API key validator initialized");
128
129    // Start HTTP server
130    let host = config.host.clone();
131    let port = config.port;
132    let workers = config.workers;
133
134    info!(
135        "Starting HTTP server on {}:{} with {} workers",
136        host, port, workers
137    );
138
139    HttpServer::new(move || {
140        App::new()
141            // Share state
142            .app_data(web::Data::new(rate_limiter.clone()))
143            .app_data(web::Data::new(api_key_validator.clone()))
144            .app_data(web::Data::new(config.clone()))
145            // Middleware
146            .wrap(tracing_actix_web::TracingLogger::default())
147            .wrap(actix_cors::Cors::permissive())
148            // Routes
149            .configure(routes::configure)
150    })
151    .workers(workers)
152    .bind((host.as_str(), port))?
153    .run()
154    .await
155    .map_err(Into::into)
156}
