1//! Middleware for actix-web
2//!
3//! Provides custom middleware for:
4//! - Request ID tracking
5//! - Request/response logging  
6//! - Metrics collection
7
8use actix_web::{
9    body::MessageBody,
10    dev::{ServiceRequest, ServiceResponse},
11    Error, HttpMessage,
12};
13use std::time::Instant;
14use tracing::{info, warn};
15use uuid::Uuid;
16
17/// Request ID header name
18pub const REQUEST_ID_HEADER: &str = "X-Request-ID";
19
20/// Request ID middleware - adds unique ID to each request
21pub struct RequestId;
22
23/// Add a unique request ID to the request
24pub fn add_request_id(req: &ServiceRequest) -> String {
25    // Check if request already has an ID
26    if let Some(id) = req.headers().get(REQUEST_ID_HEADER) {
27        if let Ok(id_str) = id.to_str() {
28            return id_str.to_string();
29        }
30    }
31
32    // Generate new request ID
33    let request_id = Uuid::new_v4().to_string();
34    req.extensions_mut().insert(request_id.clone());
35    request_id
36}
37
38/// Log request details
39pub fn log_request(req: &ServiceRequest, request_id: &str) {
40    let method = req.method().as_str();
41    let path = req.path();
42    let query = req.query_string();
43
44    let peer_addr = req
45        .peer_addr()
46        .map(|addr| addr.to_string())
47        .unwrap_or_else(|| "unknown".to_string());
48
49    if query.is_empty() {
50        info!(
51            request_id = %request_id,
52            method = %method,
53            path = %path,
54            peer = %peer_addr,
55            "Incoming request"
56        );
57    } else {
58        info!(
59            request_id = %request_id,
60            method = %method,
61            path = %path,
62            query = %query,
63            peer = %peer_addr,
64            "Incoming request"
65        );
66    }
67}
68
69/// Log response details
70pub fn log_response<B: MessageBody>(res: &ServiceResponse<B>, request_id: &str, duration_ms: f64) {
71    let status = res.status().as_u16();
72    let method = res.request().method().as_str();
73    let path = res.request().path();
74
75    if status >= 500 {
76        warn!(
77            request_id = %request_id,
78            method = %method,
79            path = %path,
80            status = %status,
81            duration_ms = %duration_ms,
82            "Request failed"
83        );
84    } else if status >= 400 {
85        warn!(
86            request_id = %request_id,
87            method = %method,
88            path = %path,
89            status = %status,
90            duration_ms = %duration_ms,
91            "Request error"
92        );
93    } else {
94        info!(
95            request_id = %request_id,
96            method = %method,
97            path = %path,
98            status = %status,
99            duration_ms = %duration_ms,
100            "Request completed"
101        );
102    }
103}
104
105/// Middleware wrapper for logging and metrics
106pub async fn logging_middleware(
107    req: ServiceRequest,
108    next: impl Fn(ServiceRequest) -> Result<ServiceResponse, Error>,
109) -> Result<ServiceResponse, Error> {
110    let start = Instant::now();
111    let request_id = add_request_id(&req);
112
113    log_request(&req, &request_id);
114
115    let res = next(req)?;
116
117    let duration = start.elapsed();
118    let duration_ms = duration.as_secs_f64() * 1000.0;
119
120    log_response(&res, &request_id, duration_ms);
121
122    Ok(res)
123}

125#[cfg(test)]
126mod tests {
127    use super::*;
128    use actix_web::test::TestRequest;
129
130    #[test]
131    fn test_request_id_generation() {
132        let id1 = Uuid::new_v4().to_string();
133        let id2 = Uuid::new_v4().to_string();
134        assert_ne!(id1, id2);
135        assert_eq!(id1.len(), 36); // UUID v4 format
136    }
137
138    #[test]
139    fn test_add_request_id_uses_existing_header() {
140        let req = TestRequest::default()
141            .insert_header((REQUEST_ID_HEADER, "existing-id"))
142            .to_srv_request();
143
144        let request_id = add_request_id(&req);
145
146        assert_eq!(request_id, "existing-id");
147        assert!(req.extensions().get::<String>().is_none());
148    }
149
150    #[test]
151    fn test_add_request_id_sets_extension() {
152        let req = TestRequest::default().to_srv_request();
153
154        let request_id = add_request_id(&req);
155        let stored = req.extensions().get::<String>().cloned();
156
157        assert!(stored.is_some());
158        assert_eq!(stored.unwrap(), request_id);
159    }
