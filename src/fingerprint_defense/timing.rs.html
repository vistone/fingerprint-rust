1//! Timing protection module
2
3// Timing anomaly detection thresholds
4/// Standard deviation threshold in milliseconds for detecting bot-like timing consistency
5const CONSISTENCY_THRESHOLD_MS: f64 = 5.0;
6
7/// Minimum interval in milliseconds for human interaction - faster intervals are suspicious
8const MIN_HUMAN_INTERVAL_MS: u64 = 50;
9
10/// Maximum number of identical timing intervals before considering it automated behavior
11const MAX_IDENTICAL_INTERVALS: usize = 3;
12
13/// Timestamp precision for hiding timing resolution (milliseconds)
14const TIMESTAMP_PRECISION_MS: u64 = 100;
15
16/// Timing protector for adding random delays and obfuscation
17pub struct TimingProtector {
18    min_delay_ms: u64,
19    max_delay_ms: u64,
20}
21
22impl TimingProtector {
23    /// Create new timing protector
24    pub fn new(min_delay_ms: u64, max_delay_ms: u64) -> Self {
25        TimingProtector {
26            min_delay_ms,
27            max_delay_ms,
28        }
29    }
30
31    /// Add random delay to response
32    pub async fn add_random_delay(&self) {
33        use rand::Rng;
34        let delay = rand::thread_rng().gen_range(self.min_delay_ms..=self.max_delay_ms);
35        tokio::time::sleep(tokio::time::Duration::from_millis(delay)).await;
36    }
37
38    /// Hide timing resolution
39    pub fn hide_timing_resolution(&self, timestamp: u64) -> u64 {
40        // Round timestamp to reduce precision
41        (timestamp / TIMESTAMP_PRECISION_MS) * TIMESTAMP_PRECISION_MS
42    }
43
44    /// Detect timing anomalies
45    ///
46    /// Analyzes a sequence of timestamps to detect suspicious timing patterns
47    /// that may indicate automated or scripted behavior.
48    ///
49    /// # Detection methods
50    /// - Checks for abnormally consistent intervals (bot-like behavior)
51    /// - Detects intervals that are too fast for human interaction
52    /// - Identifies suspicious regularity in timing patterns
53    pub fn detect_timing_anomalies(&self, timestamps: &[u64]) -> bool {
54        if timestamps.len() < 3 {
55            // Not enough data to detect anomalies
56            return false;
57        }
58
59        // Calculate intervals between consecutive timestamps
60        let mut intervals: Vec<u64> = timestamps
61            .windows(2)
62            .map(|w| w[1].saturating_sub(w[0]))
63            .collect();
64
65        if intervals.is_empty() {
66            return false;
67        }
68
69        // Check 1: Detect suspiciously consistent intervals (bot-like behavior)
70        let mean = intervals.iter().sum::<u64>() as f64 / intervals.len() as f64;
71        let variance: f64 = intervals
72            .iter()
73            .map(|&x| {
74                let diff = x as f64 - mean;
75                diff * diff
76            })
77            .sum::<f64>()
78            / intervals.len() as f64;
79        let std_dev = variance.sqrt();
80
81        // If standard deviation is very low, intervals are too consistent
82        if std_dev < CONSISTENCY_THRESHOLD_MS && mean > 0.0 {
83            return true;
84        }
85
86        // Check 2: Detect intervals that are too fast (< MIN_HUMAN_INTERVAL_MS is suspicious)
87        let fast_intervals = intervals
88            .iter()
89            .filter(|&&interval| interval < MIN_HUMAN_INTERVAL_MS && interval > 0)
90            .count();
91
92        if fast_intervals > intervals.len() / 2 {
93            // More than half of intervals are suspiciously fast
94            return true;
95        }
96
97        // Check 3: Detect exact interval repetitions (highly suspicious)
98        intervals.sort_unstable();
99        let mut repetition_count = 1;
100        let mut max_repetitions = 1;
101
102        for i in 1..intervals.len() {
103            if intervals[i] == intervals[i - 1] && intervals[i] > 0 {
104                repetition_count += 1;
105                max_repetitions = max_repetitions.max(repetition_count);
106            } else {
107                repetition_count = 1;
108            }
109        }
110
111        // If more than MAX_IDENTICAL_INTERVALS, likely automated
112        if max_repetitions > MAX_IDENTICAL_INTERVALS {
113            return true;
114        }
115
116        false
117    }
118}
