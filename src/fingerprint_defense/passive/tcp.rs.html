29/// TCP fingerprint
30#[derive(Debug, Clone, Serialize, Deserialize)]
31pub struct TcpFingerprint {
32    /// matchsignature
33    pub signature: Option<TcpSignature>,
34
35    /// similardegree
36    pub similarity: f64,
37
38    /// detect to operating system
39    pub os: Option<String>,
40
41    /// originalbeginningtrait
42    pub features: TcpFeatures,
43
44    /// fingerprintmetadata
45    pub metadata: fingerprint_core::metadata::FingerprintMetadata,
46}
47
48impl fingerprint_core::fingerprint::Fingerprint for TcpFingerprint {
49    fn fingerprint_type(&self) -> fingerprint_core::fingerprint::FingerprintType {
50        fingerprint_core::fingerprint::FingerprintType::Tcp
51    }
52
53    fn id(&self) -> String {
54        self.signature
55            .as_ref()
56            .map(|s| s.id.clone())
57            .unwrap_or_else(|| "unknown".to_string())
58    }
59
60    fn metadata(&self) -> &fingerprint_core::metadata::FingerprintMetadata {
61        &self.metadata
62    }
63
64    fn metadata_mut(&mut self) -> &mut fingerprint_core::metadata::FingerprintMetadata {
65        &mut self.metadata
66    }
67
68    fn hash(&self) -> u64 {
69        use std::collections::hash_map::DefaultHasher;
70        use std::hash::{Hash, Hasher};
71        let mut hasher = DefaultHasher::new();
72        self.id().hash(&mut hasher);
73        hasher.finish()
74    }
75
76    fn similar_to(&self, other: &dyn fingerprint_core::fingerprint::Fingerprint) -> bool {
77        if other.fingerprint_type() != fingerprint_core::fingerprint::FingerprintType::Tcp {
78            return false;
79        }
80        self.id() == other.id()
81    }
82
83    fn to_string(&self) -> String {
84        format!(
85            "TCP Fingerprint (ID: {}, Similarity: {:.2})",
86            self.id(),
87            self.similarity
88        )
89    }
90}

192    /// analysis TCP countpacket
193    pub fn analyze(&self, packet: &Packet) -> Option<TcpFingerprint> {
194        let tcp_header = packet.tcp_header.as_ref()?;
195
196        // Extract TCP trait
197        let features = self.extract_features(packet, tcp_header);
198
199        // matchsignature
200        let (signature, similarity) = self.match_signature(&features);
201
202        let mut metadata = fingerprint_core::metadata::FingerprintMetadata::new();
203
204        if let Some(os_type) = signature.as_ref().and_then(|sig| sig.os_type.clone()) {
205            metadata.set("os", &os_type);
206        }
207
208        // Calculate JA4T
209        let ja4t = fingerprint_core::ja4::JA4T::generate(
210            features.window,
211            &features.options_str,
212            features.mss.unwrap_or(0),
213            features.ttl,
214        );
215        metadata.add_tag(format!("ja4t:{}", ja4t));
216
217        Some(TcpFingerprint {
218            signature: signature.clone(),
219            similarity,
220            os: signature.as_ref().and_then(|s| s.os_type.clone()),
221            features,
222            metadata,
223        })
224    }

226    /// Extract TCP trait
227    fn extract_features(&self, packet: &Packet, tcp_header: &TcpHeader) -> TcpFeatures {
228        // inferinitialbeginning TTL
229        let initial_ttl = self.infer_initial_ttl(packet.ttl);
230
231        // Extract MSS
232        let mss = self.extract_mss(&tcp_header.options);
233
234        // Extract Window Scale
235        let window_scale = self.extract_window_scale(&tcp_header.options);
236
237        // Generateoptionsstring
238        let options_str = self.build_options_string(&tcp_header.options);
239
240        TcpFeatures {
241            ttl: packet.ttl,
242            initial_ttl,
243            window: tcp_header.window,
244            mss,
245            window_scale,
246            options_str,
247            ip_flags: packet.ip_flags,
248        }
249    }

251    /// inferinitialbeginning TTL
252    fn infer_initial_ttl(&self, ttl: u8) -> u8 {
253        // Based onobserve to TTL inferinitialbeginning TTL
254        // commoninitialbeginning TTL value：32, 64, 128, 255
255        if ttl <= 32 {
256            32
257        } else if ttl <= 64 {
258            64
259        } else if ttl <= 128 {
260            128
261        } else {
262            255
263        }
264    }

266    /// Extract MSS
267    fn extract_mss(&self, options: &[crate::passive::packet::TcpOption]) -> Option<u16> {
268        for opt in options {
269            if opt.kind() == 2 && opt.data().len() >= 2 {
270                // MSS option
271                let data = opt.data();
272                return Some(u16::from_be_bytes([data[0], data[1]]));
273            }
274        }
275        None
276    }
277
278    /// Extract Window Scale
279    fn extract_window_scale(&self, options: &[crate::passive::packet::TcpOption]) -> Option<u8> {
280        for opt in options {
281            if opt.kind() == 3 && !opt.data().is_empty() {
282                // Window Scale option
283                let data = opt.data();
284                return Some(data[0]);
285            }
286        }
287        None
288    }

290    /// Buildoptionsstring
291    fn build_options_string(&self, options: &[crate::passive::packet::TcpOption]) -> String {
292        let mut parts = Vec::new();
293
294        for opt in options {
295            match opt.kind() {
296                0 => break,                          // End of options
297                1 => parts.push("nop".to_string()),  // NOP
298                2 => parts.push("mss".to_string()),  // MSS
299                3 => parts.push("ws".to_string()),   // Window Scale
300                4 => parts.push("sack".to_string()), // SACK permitted
301                8 => parts.push("ts".to_string()),   // Timestamp
302                _ => parts.push(format!("opt{}", opt.kind())),
303            }
304        }
305
306        parts.join(",")
307    }

309    /// matchsignature
310    fn match_signature(&self, features: &TcpFeatures) -> (Option<TcpSignature>, f64) {
311        let mut best_match: Option<(&TcpSignature, f64)> = None;
312
313        for sig in self.signatures.values() {
314            let similarity = self.calculate_similarity(features, sig);
315
316            if let Some((_, best_sim)) = best_match {
317                if similarity > best_sim {
318                    best_match = Some((sig, similarity));
319                }
320            } else {
321                best_match = Some((sig, similarity));
322            }
323        }
324
325        if let Some((sig, sim)) = best_match {
326            if sim > 0.6 {
327                // similarthresholdvalue
328                (Some(sig.clone()), sim)
329            } else {
330                (None, sim)
331            }
332        } else {
333            (None, 0.0)
334        }
335    }

337    /// Calculatesimilardegree
338    fn calculate_similarity(&self, features: &TcpFeatures, signature: &TcpSignature) -> f64 {
339        let mut score = 0.0;
340        let mut total = 0.0;
341
342        // TTL match
343        if signature.ttl > 0 {
344            total += 1.0;
345            let ttl_diff = (features.initial_ttl as i16 - signature.ttl as i16).abs();
346            if ttl_diff == 0 {
347                score += 1.0;
348            } else if ttl_diff <= 1 {
349                score += 0.8;
350            } else {
351                score += 0.5;
352            }
353        }
354
355        // Window Size match (simplify)
356        if signature.window_size > 0 {
357            total += 1.0;
358            let window_diff = (features.window as i32 - signature.window_size as i32).abs();
359            if window_diff < 100 {
360                score += 1.0;
361            } else if window_diff < 1000 {
362                score += 0.7;
363            } else {
364                score += 0.3;
365            }
366        }
367
368        // MSS match
369        if let Some(sig_mss) = signature.mss {
370            total += 1.0;
371            if let Some(feat_mss) = features.mss {
372                if sig_mss == feat_mss {
373                    score += 1.0;
374                } else {
375                    let mss_diff = (sig_mss as i32 - feat_mss as i32).abs();
376                    if mss_diff < 10 {
377                        score += 0.8;
378                    } else {
379                        score += 0.4;
380                    }
381                }
382            }
383        }
384
385        // Window Scale match
386        if let Some(sig_ws) = signature.window_scale {
387            total += 1.0;
388            if let Some(feat_ws) = features.window_scale {
389                if sig_ws == feat_ws {
390                    score += 1.0;
391                } else {
392                    score += 0.5;
393                }
394            }
395        }
396
397        if total > 0.0 {
398            score / total
399        } else {
400            0.0
401        }
402    }
