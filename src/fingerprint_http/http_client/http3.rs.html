28#[cfg(feature = "http3")]
29pub fn send_http3_request(
30    host: &str,
31    port: u16,
32    path: &str,
33    request: &HttpRequest,
34    config: &HttpClientConfig,
35) -> Result<HttpResponse> {
36    // Fix: useglobalsingleton Runtime, avoideach timerequest都Create a newrun when
37    RUNTIME.block_on(async { send_http3_request_async(host, port, path, request, config).await })
38}

40#[cfg(feature = "http3")]
41async fn send_http3_request_async(
42    host: &str,
43    port: u16,
44    path: &str,
45    request: &HttpRequest,
46    config: &HttpClientConfig,
47) -> Result<HttpResponse> {
48    use std::net::{IpAddr, Ipv4Addr, Ipv6Addr, SocketAddr, ToSocketAddrs};
49    use std::sync::Arc;
50    use std::time::{Duration, Instant};
51
52    let start = Instant::now();
53
54    // 1. configuration QUIC client
55    let tls_config = super::rustls_utils::build_client_config(
56        config.verify_tls,
57        vec![b"h3".to_vec()],
58        config.profile.as_ref(),
59    );
60
61    let mut client_config = ClientConfig::new(Arc::new(
62        quinn::crypto::rustls::QuicClientConfig::try_from(tls_config).map_err(|e| {
63            HttpClientError::TlsError(format!("Failed to create QUIC config: {}", e))
64        })?,
65    ));
66
67    // optimizetransferconfigurationending withimproveperformance and connectionmigratecapability
68    let mut transport = TransportConfig::default();
69
70    // connectionmigrate (Connection Migration) optimize
71    // QUIC allow in IP toggle when keepconnection, thispairmobilesimulate至closeimportant
72    transport.initial_rtt(Duration::from_millis(100));
73    transport.max_idle_timeout(Some(Duration::from_secs(60).try_into().map_err(|e| {
74        HttpClientError::ConnectionFailed(format!("configurationtimeoutfailure: {}", e))
75    })?));
76    // increasekeep-alivefrequencyending withauxiliaryconnectionmigrateidentify
77    transport.keep_alive_interval(Some(Duration::from_secs(20)));
78
79    // allowpairendmigrate (defaultalreadyopen, hereexplicitexplain其importantproperty)
80    // transport.allow_peer_migration(true);
81
82    // simulate Chrome streamcontrolwindow (Chrome usuallyuselargerwindowending withimprove吞吐)
83    transport.stream_receive_window((6 * 1024 * 1024u32).into()); // 6MB (Chrome style)
84    transport.receive_window((15 * 1024 * 1024u32).into()); // 15MB (Chrome style)
85
86    // allowmoreconcurrentstream
87    transport.max_concurrent_bidi_streams(100u32.into());
88    transport.max_concurrent_uni_streams(100u32.into());
89
90    client_config.transport_config(Arc::new(transport));
91
92    // 2. DNS Parse (priority IPv4, avoid IPv4 endpoint connection IPv6 remote cause invalid remote address)
93    let addr_str = format!("{}:{}", host, port);
94    let mut addrs: Vec<SocketAddr> = addr_str
95        .to_socket_addrs()
96        .map_err(|e| HttpClientError::InvalidUrl(format!("DNS Parsefailure: {}", e)))?
97        .collect();
98    if addrs.is_empty() {
99        return Err(HttpClientError::InvalidUrl(
100            "unable toParseaddress".to_string(),
101        ));
102    }
103    addrs.sort_by_key(|a| matches!(a.ip(), IpAddr::V6(_))); // IPv4 priority
104
105    // 4. connection to server (Happy Eyeballs simplify版：looptryallParse to address)
106    let mut connection_result = Err(HttpClientError::ConnectionFailed(
107        "无availableaddress".to_string(),
108    ));
109
110    for remote_addr in addrs {
111        // Create QUIC endpoint ( by remote address族selectbind)
112        let bind_addr = match remote_addr.ip() {
113            IpAddr::V4(_) => SocketAddr::new(IpAddr::V4(Ipv4Addr::UNSPECIFIED), 0),
114            IpAddr::V6(_) => SocketAddr::new(IpAddr::V6(Ipv6Addr::UNSPECIFIED), 0),
115        };
116
117        let endpoint = match Endpoint::client(bind_addr) {
118            Ok(mut ep) => {
119                ep.set_default_client_config(client_config.clone());
120                ep
121            }
122            Err(_) => continue,
123        };
124
125        match endpoint.connect(remote_addr, host) {
126            Ok(connecting) => {
127                match connecting.await {
128                    Ok(conn) => {
129                        // 5. establish HTTP/3 connection
130                        match h3::client::new(h3_quinn::Connection::new(conn)).await {
131                            Ok((driver, send_request)) => {
132                                connection_result = Ok((driver, send_request));
133                                break;
134                            }
135                            Err(e) => {
136                                connection_result = Err(HttpClientError::ConnectionFailed(
137                                    format!("HTTP/3 handshakefailure: {}", e),
138                                ));
139                            }
140                        }
141                    }
142                    Err(e) => {
143                        connection_result = Err(HttpClientError::ConnectionFailed(format!(
144                            "QUIC handshakefailure: {}",
145                            e
146                        )));
147                    }
148                }
149            }
150            Err(e) => {
151                connection_result = Err(HttpClientError::ConnectionFailed(format!(
152                    "QUIC connection发起failure: {}",
153                    e
154                )));
155            }
156        }
157    }
158
159    let (driver, mut send_request) = connection_result?;
160
161    // in backdriverconnection：h3 0.0.4 driver need被continuous poll_close
162    tokio::spawn(async move {
163        let mut driver = driver;
164        let _ = std::future::poll_fn(|cx| driver.poll_close(cx)).await;
165    });
166
167    // 6. Buildrequest
168    let uri = format!("https://{}{}", host, path);
169    let mut http_request = http::Request::builder()
170        .method(request.method.as_str())
171        .uri(uri)
172        .version(http::Version::HTTP_3);
173
174    // Fix: Add Cookie to request ( if exists)
175    let mut request_with_cookies = request.clone();
176    if let Some(cookie_store) = &config.cookie_store {
177        super::request::add_cookies_to_request(
178            &mut request_with_cookies,
179            cookie_store,
180            host,
181            path,
182            true, // HTTPS is securityconnection
183        );
184    }
185
186    // Add headers
187    // Note: do notmanualAdd host header, h3 willautomatic from URI Extract
188    http_request = http_request.header("user-agent", &config.user_agent);
189
190    for (key, value) in &request_with_cookies.headers {
191        // skip host header ( if userpassed in)
192        if key.to_lowercase() != "host" {
193            http_request = http_request.header(key, value);
194        }
195    }
196
197    // Fix: Buildrequest (h3 need Request<()>, thenthrough stream send body)
198    let http_request = http_request
199        .body(())
200        .map_err(|e| HttpClientError::InvalidResponse(format!("Buildrequestfailure: {}", e)))?;
201
202    // 7. sendrequest
203    let mut stream = send_request
204        .send_request(http_request)
205        .await
206        .map_err(|e| HttpClientError::ConnectionFailed(format!("sendrequestfailure: {}", e)))?;
207
208    // Fix: through stream sendrequest体 ( if exists)
209    if let Some(body) = &request.body {
210        if !body.is_empty() {
211            stream
212                .send_data(bytes::Bytes::from(body.clone()))
213                .await
214                .map_err(|e| {
215                    HttpClientError::ConnectionFailed(format!("Failed to send request body: {}", e))
216                })?;
217        }
218    }
219
220    stream
221        .finish()
222        .await
223        .map_err(|e| HttpClientError::ConnectionFailed(format!("endrequestfailure: {}", e)))?;
224
225    // 8. receiveresponse
226    let response = stream
227        .recv_response()
228        .await
229        .map_err(|e| HttpClientError::InvalidResponse(format!("receiveresponsefailure: {}", e)))?;
230
231    let status_code = response.status().as_u16();
232    let headers = response.headers().clone();
233
234    // securityFix: Check HTTP/3 responseheadersize, prevent QPACK compressionbombattack
235    // h3 crate 0.0.4 defaultlimitusuallylarger, weAdd额outsideCheck
236    const MAX_HTTP3_HEADER_SIZE: usize = 64 * 1024; // 64KB (RFC 9114 suggestminimumvalue)
237    let total_header_size: usize = headers
238        .iter()
239        .map(|(k, v)| k.as_str().len() + v.len())
240        .sum();
241    if total_header_size > MAX_HTTP3_HEADER_SIZE {
242        return Err(HttpClientError::InvalidResponse(format!(
243            "HTTP/3 responseheadertoo large (>{} bytes)",
244            MAX_HTTP3_HEADER_SIZE
245        )));
246    }
247
248    // receive body
249    use bytes::Buf;
250    let mut body_data = Vec::new();
251
252    // securitylimit：prevent HTTP/3 responsebody too largecauseinsidememory exhausted
253    const MAX_HTTP3_BODY_SIZE: usize = 100 * 1024 * 1024; // 100MB
254
255    while let Some(mut chunk) = stream.recv_data().await.map_err(|e| {
256        HttpClientError::Io(std::io::Error::other(format!("read body failure: {}", e)))
257    })? {
258        // use Buf trait readcountdata
259        let chunk_len = chunk.remaining();
260
261        // securityCheck：preventresponsebody too large
262        if body_data.len().saturating_add(chunk_len) > MAX_HTTP3_BODY_SIZE {
263            return Err(HttpClientError::InvalidResponse(format!(
264                "HTTP/3 responsebody too large (>{} bytes)",
265                MAX_HTTP3_BODY_SIZE
266            )));
267        }
268
269        let mut chunk_bytes = vec![0u8; chunk_len];
270        chunk.copy_to_slice(&mut chunk_bytes);
271        body_data.extend_from_slice(&chunk_bytes);
272    }
273
274    let elapsed = start.elapsed().as_millis() as u64;
275
276    // 9. Buildresponse
277    let mut response_headers = std::collections::HashMap::new();
278    for (key, value) in headers.iter() {
279        if let Ok(value_str) = value.to_str() {
280            response_headers.insert(key.as_str().to_lowercase(), value_str.to_string());
281        }
282    }
283
284    Ok(HttpResponse {
285        status_code,
286        status_text: http::StatusCode::from_u16(status_code)
287            .map(|s| s.canonical_reason().unwrap_or("Unknown").to_string())
288            .unwrap_or_else(|_| "Unknown".to_string()),
289        headers: response_headers,
290        body: body_data,
291        http_version: "HTTP/3".to_string(),
292        response_time_ms: elapsed,
293    })
294}
