1//! rustls configuration utilities (provides HTTP/1/2/3 reuse)
2//!
3//! This module provides utility functions for building rustls configurations.
4//!
5//! ## Features
6//!
7//! - `build_root_store()`: Build root certificate store using Mozilla roots
8//! - `apply_verify_tls()`: Configure TLS certificate verification
9//! - `build_client_config()`: Build complete rustls ClientConfig with ALPN and verification
10//!
11//! ## Security Warning
12//!
13//! **The `dangerous_configuration` feature allows disabling TLS certificate verification.**
14//!
15//! **DO NOT USE IN PRODUCTION!** This feature is intended only for:
16//! - Local development and testing
17//! - Internal network environments where certificates cannot be properly issued
18//! - Debugging TLS handshake issues
19//!
20//! Disabling certificate verification exposes your application to:
21//! - Man-in-the-middle (MITM) attacks
22//! - Credential theft
23//! - Data interception
24//!
25//! If you need to work with self-signed certificates in production, consider:
26//! - Using certificate pinning instead
27//! - Adding custom root certificates to the trust store
28//! - Using proper PKI infrastructure
29

43/// Ensure the crypto provider is installed (ring)
44static INIT_CRYPTO_PROVIDER: Once = Once::new();
45
46/// Initialize the rustls crypto provider (ring) if not already done.
47/// This must be called before any TLS operations.
48fn ensure_crypto_provider() {
49    INIT_CRYPTO_PROVIDER.call_once(|| {
50        let _ = rustls::crypto::ring::default_provider().install_default();
51    });
52}

54/// Build rustls rootcertificatestore (Mozilla roots)
55pub fn build_root_store() -> rustls::RootCertStore {
56    ensure_crypto_provider();
57    let mut root_store = rustls::RootCertStore::empty();
58    root_store.extend(webpki_roots::TLS_SERVER_ROOTS.iter().cloned());
59    root_store
60}

62/// If verify_tls=false, install an "accept all certificates" verifier (dangerous feature, only for debug)
63///
64/// # Security Warning
65///
66/// **This function can completely disable TLS certificate verification when the
67/// `dangerous_configuration` feature is enabled and `verify_tls` is set to `false`.**
68///
69/// This makes your application vulnerable to:
70/// - Man-in-the-middle (MITM) attacks
71/// - Credential theft and data interception
72/// - Impersonation attacks
73///
74/// # Arguments
75///
76/// * `cfg` - The rustls ClientConfig to modify
77/// * `verify_tls` - If true, certificates are verified normally. If false and
78///   the `dangerous_configuration` feature is enabled, all certificates are accepted.
79///
80/// # Example
81///
82/// ```ignore
83/// // DO NOT USE IN PRODUCTION
84/// apply_verify_tls(&mut config, false);
85/// ```
86#[allow(unused_variables)]
87pub fn apply_verify_tls(cfg: &mut rustls::ClientConfig, verify_tls: bool) {
88    if verify_tls {
89        return;
90    }
91
92    // Note: rustls 0.23 API changed - dangerous features now under danger module
93    // If verify_tls=false, use dangerous configuration to accept all certificates.
94    // This needs the rustls dangerous_configuration feature.
95    #[cfg(feature = "dangerous_configuration")]
96    {
97        use rustls::client::danger::{
98            HandshakeSignatureValid, ServerCertVerified, ServerCertVerifier,
99        };
100        use rustls::pki_types::{CertificateDer, ServerName, UnixTime};
101        use rustls::{DigitallySignedStruct, Error as RustlsError, SignatureScheme};
102
103        #[derive(Debug)]
104        struct NoCertificateVerification;
105
106        impl ServerCertVerifier for NoCertificateVerification {
107            fn verify_server_cert(
108                &self,
109                _end_entity: &CertificateDer,
110                _intermediates: &[CertificateDer],
111                _server_name: &ServerName,
112                _ocsp_response: &[u8],
113                _now: UnixTime,
114            ) -> std::result::Result<ServerCertVerified, RustlsError> {
115                Ok(ServerCertVerified::assertion())
116            }
117
118            fn verify_tls12_signature(
119                &self,
120                _message: &[u8],
121                _cert: &CertificateDer,
122                _dss: &DigitallySignedStruct,
123            ) -> std::result::Result<HandshakeSignatureValid, RustlsError> {
124                Ok(HandshakeSignatureValid::assertion())
125            }
126
127            fn verify_tls13_signature(
128                &self,
129                _message: &[u8],
130                _cert: &CertificateDer,
131                _dss: &DigitallySignedStruct,
132            ) -> std::result::Result<HandshakeSignatureValid, RustlsError> {
133                Ok(HandshakeSignatureValid::assertion())
134            }
135
136            fn supported_verify_schemes(&self) -> Vec<SignatureScheme> {
137                vec![
138                    SignatureScheme::RSA_PKCS1_SHA1,
139                    SignatureScheme::ECDSA_SHA1_Legacy,
140                    SignatureScheme::RSA_PKCS1_SHA256,
141                    SignatureScheme::ECDSA_NISTP256_SHA256,
142                    SignatureScheme::RSA_PKCS1_SHA384,
143                    SignatureScheme::ECDSA_NISTP384_SHA384,
144                    SignatureScheme::RSA_PKCS1_SHA512,
145                    SignatureScheme::ECDSA_NISTP521_SHA512,
146                    SignatureScheme::RSA_PSS_SHA256,
147                    SignatureScheme::RSA_PSS_SHA384,
148                    SignatureScheme::RSA_PSS_SHA512,
149                    SignatureScheme::ED25519,
150                    SignatureScheme::ED448,
151                ]
152            }
153        }
154
155        cfg.dangerous()
156            .set_certificate_verifier(Arc::new(NoCertificateVerification));
157    }
158
159    #[cfg(not(feature = "dangerous_configuration"))]
160    {
161        // If dangerous_configuration feature is not enabled, ignore verify_tls=false setting
162        // and always validate certificates (more secure)
163        eprintln!("warning: verify_tls=false requires dangerous_configuration feature, ignoring");
164    }
165}
