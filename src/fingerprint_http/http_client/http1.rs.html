1//! HTTP/1.1 implement
2//!
3//! use netconnpool manage TCP connection, send HTTP/1.1 request
4
5use super::{HttpClientConfig, HttpClientError, HttpRequest, HttpResponse, Result};
6use std::io::Write;
7use std::net::TcpStream;
8
9/// send HTTP/1.1 request
10pub fn send_http1_request(
11    host: &str,
12    port: u16,
13    path: &str,
14    request: &HttpRequest,
15    config: &HttpClientConfig,
16) -> Result<HttpResponse> {
17    // connectionserver
18    let addr = format!("{}:{}", host, port);
19    let mut stream = TcpStream::connect(&addr).map_err(|e| {
20        HttpClientError::ConnectionFailed(format!("Connection failed {}: {}", addr, e))
21    })?;
22
23    // settingstimeout
24    stream
25        .set_read_timeout(Some(config.read_timeout))
26        .map_err(HttpClientError::Io)?;
27    stream
28        .set_write_timeout(Some(config.write_timeout))
29        .map_err(HttpClientError::Io)?;
30
31    // Fix: Add Cookie to request ( if exists)
32    let mut request_with_cookies = request.clone();
33    if let Some(cookie_store) = &config.cookie_store {
34        super::request::add_cookies_to_request(
35            &mut request_with_cookies,
36            cookie_store,
37            host,
38            path,
39            false, // HTTP is notsecurityconnection
40        );
41    }
42
43    // Build并send HTTP/1.1 request
44    // useChromestandardheader顺序configure
45    let header_order = Some(fingerprint_headers::chrome_header_order());
46    let http_request =
47        request_with_cookies.build_http1_request_bytes(host, path, header_order.as_deref());
48    stream
49        .write_all(&http_request)
50        .map_err(HttpClientError::Io)?;
51    stream.flush().map_err(HttpClientError::Io)?;
52
53    // readresponse
54    let mut stream = stream;
55    let buffer =
56        super::io::read_http1_response_bytes(&mut stream, super::io::DEFAULT_MAX_RESPONSE_BYTES)
57            .map_err(HttpClientError::Io)?;
58
59    // Parseresponse
60    HttpResponse::parse(&buffer).map_err(HttpClientError::InvalidResponse)
61}
