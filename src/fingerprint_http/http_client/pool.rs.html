127    /// Get or Createconnection pool
128    #[cfg(feature = "connection-pool")]
129    pub fn get_pool(&self, host: &str, port: u16) -> Result<Arc<Pool>> {
130        let key = format!("{}:{}", host, port);
131        let mut pools = self.pools.lock().map_err(|e| {
132            HttpClientError::ConnectionFailed(format!("connection poollockfailure: {}", e))
133        })?;
134
135        if let Some(pool) = pools.get(&key) {
136            return Ok(pool.clone());
137        }
138
139        // Create a newconnection pool
140        let pool_config = self.create_pool_config(host, port);
141        let pool = Pool::new(pool_config).map_err(|e| {
142            HttpClientError::ConnectionFailed(format!("Createconnection poolfailure: {:?}", e))
143        })?;
144
145        let pool = Arc::new(pool);
146        pools.insert(key, pool.clone());
147
148        Ok(pool)
149    }

159    /// Createconnection poolconfiguration
160    #[cfg(feature = "connection-pool")]
161    fn create_pool_config(&self, host: &str, port: u16) -> PoolConfig {
162        let host = host.to_string();
163        let connect_timeout = self.config.connect_timeout;
164
165        PoolConfig {
166            mode: netconnpool::PoolMode::Client,
167            max_connections: self.config.max_connections,
168            min_connections: self.config.min_idle,
169            max_idle_connections: self.config.max_connections,
170            connection_timeout: self.config.connect_timeout,
171            idle_timeout: self.config.idle_timeout,
172            max_lifetime: self.config.max_lifetime,
173            get_connection_timeout: self.config.connect_timeout,
174            health_check_interval: Duration::from_secs(30),
175            health_check_timeout: Duration::from_secs(3),
176            connection_leak_timeout: Duration::from_secs(300),
177
178            // provide Dialer functionfromCreate TCP connection
179            // Note: hereunable todirectlyaccess config.profile, because dialer is closepackage
180            // TCP Profile should in Createconnection poolbeforethenapplication to config in
181            dialer: Some(Box::new(move |_protocol| {
182                use std::net::{IpAddr, SocketAddr, ToSocketAddrs};
183
184                let addrs: Vec<SocketAddr> = (host.as_str(), port)
185                    .to_socket_addrs()
186                    .map_err(|e| Box::new(e) as Box<dyn std::error::Error + Send + Sync>)?
187                    .collect();
188
189                // priorityuse IPv4, avoid in "none IPv6 route"environment in appear `Network is unreachable`.
190                let mut v4 = Vec::new();
191                let mut v6 = Vec::new();
192                for a in addrs {
193                    match a.ip() {
194                        IpAddr::V4(_) => v4.push(a),
195                        IpAddr::V6(_) => v6.push(a),
196                    }
197                }
198
199                let mut last_err: Option<std::io::Error> = None;
200                for addr in v4.into_iter().chain(v6.into_iter()) {
201                    // Note: Currently using standard connect method
202                    // Future enhancement: Apply TCP profile configuration in connection pool
203                    match TcpStream::connect_timeout(&addr, connect_timeout) {
204                        Ok(s) => return Ok(ConnectionType::Tcp(s)),
205                        Err(e) => last_err = Some(e),
206                    }
207                }
208
209                Err(Box::new(
210                    last_err.unwrap_or_else(|| std::io::Error::other("no resolved addresses")),
211                )
212                    as Box<dyn std::error::Error + Send + Sync>)
213            })),
214            listener: None,
215            acceptor: None,
216            health_checker: None,
217            close_conn: None,
218            on_created: None,
219            on_borrow: None,
220            on_return: None,
221            enable_stats: true,
222            enable_health_check: true,
223            clear_udp_buffer_on_return: false,
224            max_buffer_clear_packets: 0,
225            udp_buffer_clear_timeout: Duration::from_secs(0),
226        }
227    }

229    /// Getstatisticsinfo
230    #[cfg(feature = "connection-pool")]
231    pub fn get_stats(&self) -> Vec<PoolStats> {
232        let pools = match self.pools.lock() {
233            Ok(p) => p,
234            Err(e) => {
235                eprintln!("warning: connection poollockfailure: {}", e);
236                return Vec::new();
237            }
238        };
239        pools
240            .iter()
241            .map(|(key, pool)| {
242                let stats = pool.stats();
243                PoolStats {
244                    endpoint: key.clone(),
245                    total_connections: stats.total_connections_created,
246                    active_connections: stats.current_active_connections,
247                    idle_connections: stats.current_idle_connections,
248                    total_requests: stats.total_get_requests,
249                    successful_requests: stats.successful_gets,
250                    failed_requests: stats.failed_gets,
251                }
252            })
253            .collect()
254    }

273    /// closeallconnection pool

275    pub fn shutdown(&self) {
276        if let Ok(mut pools) = self.pools.lock() {
277            for (_, pool) in pools.iter() {

279            }
280            pools.clear();

301impl PoolStats {
302    /// GetsuccessÁéá
303    pub fn success_rate(&self) -> f64 {
304        if self.total_requests == 0 {
305            return 0.0;
306        }
307        (self.successful_requests as f64 / self.total_requests as f64) * 100.0
308    }
309

313        println!(" ÊÄªconnectioncount: {}", self.total_connections);
314        println!(" activeconnection: {}", self.active_connections);
315        println!(" emptyÈó≤connection: {}", self.idle_connections);
316        println!(" ÊÄªrequestcount: {}", self.total_requests);
317        println!(" successrequest: {}", self.successful_requests);
318        println!(" failurerequest: {}", self.failed_requests);
319        println!(" successÁéá: {:.2}%", self.success_rate());
320    }
