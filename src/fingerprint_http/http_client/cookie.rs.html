5use std::collections::HashMap;

9/// Cookie
10#[derive(Debug, Clone)]
11pub struct Cookie {
12    pub name: String,
13    pub value: String,
14    pub domain: String,
15    pub path: String,
16    pub expires: Option<SystemTime>,
17    pub max_age: Option<Duration>,
18    pub secure: bool,
19    pub http_only: bool,
20    pub same_site: Option<SameSite>,
21}
22
23/// SameSite property
24#[derive(Debug, Clone, PartialEq)]
25pub enum SameSite {
26    Strict,
27    Lax,
28    None,
29}
30
31impl Cookie {
32    /// Create a new Cookie
33    pub fn new(name: String, value: String, domain: String) -> Self {
34        Self {
35            name,
36            value,
37            domain,
38            path: "/".to_string(),
39            expires: None,
40            max_age: None,
41            secure: false,
42            http_only: false,
43            same_site: None,
44        }
45    }
46
47    /// Check Cookie whetherexpire
48    pub fn is_expired(&self) -> bool {
49        if let Some(expires) = self.expires {
50            return SystemTime::now() > expires;
51        }
52        false
53    }
54
55    /// convert to HTTP Cookie headerformat
56    pub fn to_header_value(&self) -> String {
57        format!("{}={}", self.name, self.value)
58    }
59
60    /// from Set-Cookie headerParse
61    pub fn parse_set_cookie(header: &str, domain: String) -> Option<Self> {
62        let parts: Vec<&str> = header.split(';').collect();
63        if parts.is_empty() {
64            return None;
65        }
66
67        // Parse name=value
68        let name_value: Vec<&str> = parts[0].split('=').collect();
69        if name_value.len() != 2 {
70            return None;
71        }
72
73        let mut cookie = Cookie::new(
74            name_value[0].trim().to_string(),
75            name_value[1].trim().to_string(),
76            domain,
77        );
78
79        // Parseotherproperty
80        for part in &parts[1..] {
81            let part = part.trim();
82            if part.to_lowercase().starts_with("domain=") {
83                cookie.domain = part[7..].to_string();
84            } else if part.to_lowercase().starts_with("path=") {
85                cookie.path = part[5..].to_string();
86            } else if part.to_lowercase().starts_with("max-age=") {
87                if let Ok(secs) = part[8..].parse::<u64>() {
88                    cookie.max_age = Some(Duration::from_secs(secs));
89                    // let Max-Age true生effect：convert to绝pair expires ending withreuse is_expired()
90                    cookie.expires = Some(SystemTime::now() + Duration::from_secs(secs));
91                }
92            } else if part.to_lowercase() == "secure" {
93                cookie.secure = true;
94            } else if part.to_lowercase() == "httponly" {
95                cookie.http_only = true;
96            } else if part.to_lowercase().starts_with("samesite=") {
97                let value = part[9..].to_lowercase();
98                cookie.same_site = match value.as_str() {
99                    "strict" => Some(SameSite::Strict),
100                    "lax" => Some(SameSite::Lax),
101                    "none" => Some(SameSite::None),
102                    _ => None,
103                };
104            }
105        }
106
107        Some(cookie)
108    }
109}

111/// Cookie store
112#[derive(Debug, Clone)]
113pub struct CookieStore {
114    ///  by domainstore Cookie
115    cookies: Arc<Mutex<HashMap<String, Vec<Cookie>>>>,
116}
117
118impl CookieStore {
119    /// Create a new Cookie store
120    pub fn new() -> Self {
121        Self {
122            cookies: Arc::new(Mutex::new(HashMap::new())),
123        }
124    }
125
126    /// Add Cookie
127    pub fn add_cookie(&self, cookie: Cookie) {
128        if let Ok(mut cookies) = self.cookies.lock() {
129            let domain_cookies = cookies.entry(cookie.domain.clone()).or_default();
130
131            // Checkwhetheralready existssame name Cookie,  if existsthenUpdate
132            if let Some(pos) = domain_cookies.iter().position(|c| c.name == cookie.name) {
133                domain_cookies[pos] = cookie;
134            } else {
135                domain_cookies.push(cookie);
136            }
137        } else {
138            eprintln!("warning: Cookie storelockfailure，unable toAdd Cookie");
139        }
140    }
141
142    /// from Set-Cookie headerAdd Cookie
143    pub fn add_from_response(&self, set_cookie_header: &str, domain: String) {
144        if let Some(cookie) = Cookie::parse_set_cookie(set_cookie_header, domain) {
145            self.add_cookie(cookie);
146        }
147    }
148
149    /// Getspecifieddomainallvalid Cookie
150    ///
151    /// Based on RFC 6265 specificationperformdomainmatch：
152    /// - Cookie domain property (如 `.example.com`)shouldmatch `example.com` and其allchilddomain
153    /// - `example.com` Cookie shouldmatch `example.com` and `*.example.com`
154    pub fn get_cookies_for_domain(&self, domain: &str) -> Vec<Cookie> {
155        let cookies = match self.cookies.lock() {
156            Ok(c) => c,
157            Err(e) => {
158                eprintln!("warning: Cookie storelockfailure: {}", e);
159                return Vec::new();
160            }
161        };
162        let mut result = Vec::new();
163
164        let domain_lower = domain.to_lowercase();
165        for (cookie_domain, domain_cookies) in cookies.iter() {
166            let cookie_domain_lower = cookie_domain.to_lowercase();
167
168            // Fix: correctdomainmatchlogic (RFC 6265)
169            let matches = if cookie_domain_lower == domain_lower {
170                // completelymatch
171                true
172            } else if let Some(base) = cookie_domain_lower.strip_prefix('.') {
173                // Cookie domain ending with. openheader (如.example.com)
174                // shouldmatch example.com and all *.example.com
175                domain_lower == base || domain_lower.ends_with(&format!(".{}", base))
176            } else {
177                // Cookie domain not. openheader (如 example.com)
178                // shouldmatch example.com and all *.example.com
179                domain_lower == cookie_domain_lower
180                    || domain_lower.ends_with(&format!(".{}", cookie_domain_lower))
181            };
182
183            if matches {
184                for cookie in domain_cookies {
185                    if !cookie.is_expired() {
186                        result.push(cookie.clone());
187                    }
188                }
189            }
190        }
191
192        result
193    }
194
195    /// Generate Cookie header
196    ///
197    /// # Parameters
198    /// - `domain`: requestdomain
199    /// - `path`: requestpath
200    /// - `is_secure`: whether as HTTPS connection ( for Secure Cookie Check)
201    pub fn generate_cookie_header(
202        &self,
203        domain: &str,
204        path: &str,
205        is_secure: bool,
206    ) -> Option<String> {
207        let cookies = self.get_cookies_for_domain(domain);
208        if cookies.is_empty() {
209            return None;
210        }
211
212        // filterpathmatch Cookie, 并Check Secure property
213        let matching_cookies: Vec<String> = cookies
214            .iter()
215            .filter(|c| {
216                // pathmatch
217                if !path.starts_with(&c.path) {
218                    return false;
219                }
220                // securityFix: Secure Cookie can only in HTTPS connectionupsend
221                if c.secure && !is_secure {
222                    return false;
223                }
224                true
225            })
226            .map(Cookie::to_header_value)
227            .collect();
228
229        if matching_cookies.is_empty() {
230            None
231        } else {
232            Some(matching_cookies.join("; "))
233        }
234    }
235
236    /// clearspecifieddomain Cookie
237    pub fn clear_domain(&self, domain: &str) {
238        if let Ok(mut cookies) = self.cookies.lock() {
239            cookies.remove(domain);
240        }
241    }
242
243    /// clearall Cookie
244    pub fn clear_all(&self) {
245        if let Ok(mut cookies) = self.cookies.lock() {
246            cookies.clear();
247        }
248    }
249
250    /// clearexpire Cookie
251    pub fn cleanup_expired(&self) {
252        if let Ok(mut cookies) = self.cookies.lock() {
253            for domain_cookies in cookies.values_mut() {
254                domain_cookies.retain(|c| !c.is_expired());
255            }
256        }
257    }
258
259    /// Getall Cookie count
260    pub fn count(&self) -> usize {
261        self.cookies
262            .lock()
263            .map(|cookies| cookies.values().map(|v| v.len()).sum())
264            .unwrap_or(0)
265    }
266}
