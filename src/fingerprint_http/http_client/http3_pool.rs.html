1//! HTTP/3 with Connection Pool
2//!
3//! architectureexplain：
4//! - HTTP/3 adoptsessionpool (H3SessionPool)implement QUIC sessionreuse
5//! - pool化pair象：h3::client::SendRequest handle (alreadyhandshakecomplete QUIC session)
6//! - reusemethod：concurrentmultiplereuse (an QUIC connectioncan when processmultiple Stream)
7//! - QUIC Features：protocolthis身includingconnectionmigrate and statusmanage, no need netconnpool
8//! - sessionestablishback, connectionlifecycleby H3Session backbackground task (Driver)manage
9
10#[cfg(all(feature = "connection-pool", feature = "http3"))]
11use super::pool::ConnectionPoolManager;
12use super::{HttpClientConfig, HttpClientError, HttpRequest, HttpResponse, Result};
13#[cfg(all(feature = "connection-pool", feature = "http3"))]
14use std::sync::Arc;
15#[cfg(all(feature = "connection-pool", feature = "http3"))]
16use std::time::{Duration, Instant};

18/// useconnection poolsend HTTP/3 request
19#[cfg(all(feature = "connection-pool", feature = "http3"))]
20pub async fn send_http3_request_with_pool(
21    host: &str,
22    port: u16,
23    path: &str,
24    request: &HttpRequest,
25    config: &HttpClientConfig,
26    pool_manager: &Arc<ConnectionPoolManager>,
27) -> Result<HttpResponse> {
28    use bytes::Buf;
29    use h3_quinn::quinn;
30    use http::{Request as HttpRequest2, Version};
31
32    //Fix: use H3SessionPool implementtrue multiplexreuse
33    let session_pool = pool_manager.h3_session_pool();
34    let key = format!("{}:{}", host, port);
35
36    let start = Instant::now();
37
38    // Get or Createsession
39    let send_request_mutex = session_pool
40        .get_or_create_session(&key, async {
41            // Parsetargetaddress
42            use std::net::{IpAddr, Ipv4Addr, Ipv6Addr, SocketAddr, ToSocketAddrs};
43            let addr = format!("{}:{}", host, port);
44            let mut addrs: Vec<SocketAddr> = addr
45                .to_socket_addrs()
46                .map_err(|e| HttpClientError::ConnectionFailed(format!("DNS Parsefailure: {}", e)))?
47                .collect();
48            if addrs.is_empty() {
49                return Err(HttpClientError::ConnectionFailed(
50                    "DNS Parse无result".to_string(),
51                ));
52            }
53            addrs.sort_by_key(|a| matches!(a.ip(), IpAddr::V6(_))); // IPv4 priority
54            let remote_addr = addrs[0];
55
56            // Create QUIC clientconfiguration
57            let tls_config = super::rustls_utils::build_client_config(
58                config.verify_tls,
59                vec![b"h3".to_vec()],
60                config.profile.as_ref(),
61            );
62
63            let mut client_config = quinn::ClientConfig::new(std::sync::Arc::new(
64                quinn::crypto::rustls::QuicClientConfig::try_from(tls_config).map_err(|e| {
65                    HttpClientError::TlsError(format!("Failed to create QUIC config: {}", e))
66                })?,
67            ));
68
69            // optimizetransferconfigurationending withimproveperformance
70            let mut transport_config = quinn::TransportConfig::default();
71            transport_config.initial_rtt(Duration::from_millis(100));
72            transport_config.max_idle_timeout(Some(
73                quinn::IdleTimeout::try_from(std::time::Duration::from_secs(60)).map_err(|e| {
74                    HttpClientError::Http3Error(format!("configurationtimeoutfailure: {}", e))
75                })?,
76            ));
77            transport_config.keep_alive_interval(Some(Duration::from_secs(10)));
78
79            // increasereceivewindowending withimprovethroughput
80            transport_config.stream_receive_window((1024 * 1024u32).into()); // 1MB
81            transport_config.receive_window((10 * 1024 * 1024u32).into()); // 10MB
82
83            // allowmoreconcurrentstream
84            transport_config.max_concurrent_bidi_streams(100u32.into());
85            transport_config.max_concurrent_uni_streams(100u32.into());
86
87            client_config.transport_config(std::sync::Arc::new(transport_config));
88
89            // Create QUIC endpoint
90            let bind_addr = match remote_addr.ip() {
91                IpAddr::V4(_) => SocketAddr::new(IpAddr::V4(Ipv4Addr::UNSPECIFIED), 0),
92                IpAddr::V6(_) => SocketAddr::new(IpAddr::V6(Ipv6Addr::UNSPECIFIED), 0),
93            };
94            let mut endpoint = quinn::Endpoint::client(bind_addr).map_err(|e| {
95                HttpClientError::Http3Error(format!("Create endpoint failure: {}", e))
96            })?;
97            endpoint.set_default_client_config(client_config);
98
99            // connection to server
100            let connecting = endpoint
101                .connect(remote_addr, host)
102                .map_err(|e| HttpClientError::Http3Error(format!("Connection failed: {}", e)))?;
103
104            let connection = connecting.await.map_err(|e| {
105                HttpClientError::Http3Error(format!("establishConnection failed: {}", e))
106            })?;
107
108            // establish HTTP/3 connection
109            let quinn_conn = h3_quinn::Connection::new(connection);
110
111            let (driver, send_request) = h3::client::new(quinn_conn).await.map_err(|e| {
112                HttpClientError::Http3Error(format!("HTTP/3 handshakefailure: {}", e))
113            })?;
114
115            Ok((driver, send_request))
116        })
117        .await?;
118
119    // Getand exclude othersproperty地use SendRequest
120    let mut send_request = send_request_mutex.lock().await;
121
122    // Build HTTP/3 request
123    let uri: http::Uri = format!("https://{}:{}{}", host, port, path)
124        .parse()
125        .map_err(|e| HttpClientError::InvalidRequest(format!("invalid URI: {}", e)))?;
126
127    let http3_request = HttpRequest2::builder()
128        .method(match request.method {
129            super::request::HttpMethod::Get => http::Method::GET,
130            super::request::HttpMethod::Post => http::Method::POST,
131            super::request::HttpMethod::Put => http::Method::PUT,
132            super::request::HttpMethod::Delete => http::Method::DELETE,
133            super::request::HttpMethod::Head => http::Method::HEAD,
134            super::request::HttpMethod::Options => http::Method::OPTIONS,
135            super::request::HttpMethod::Patch => http::Method::PATCH,
136        })
137        .uri(uri)
138        .version(Version::HTTP_3)
139        // do notmanualAdd host header, h3 willautomatic from URI Extract
140        .header("user-agent", &config.user_agent);
141
142    // Fix: Add Cookie to request ( if exists)
143    let mut request_with_cookies = request.clone();
144    if let Some(cookie_store) = &config.cookie_store {
145        super::request::add_cookies_to_request(
146            &mut request_with_cookies,
147            cookie_store,
148            host,
149            path,
150            true, // HTTPS is securityconnection
151        );
152    }
153
154    let http3_request = request_with_cookies
155        .headers
156        .iter()
157        // skip host header
158        .filter(|(k, _)| k.to_lowercase() != "host")
159        .fold(http3_request, |builder, (k, v)| builder.header(k, v));
160
161    // Fix: Buildrequest (h3 need Request<()>, thenthrough stream send body)
162    let http3_request = http3_request
163        .body(())
164        .map_err(|e| HttpClientError::InvalidRequest(format!("Buildrequestfailure: {}", e)))?;
165
166    // sendrequest
167    let mut stream = send_request
168        .send_request(http3_request)
169        .await
170        .map_err(|e| HttpClientError::Http3Error(format!("sendrequestfailure: {}", e)))?;
171
172    // Fix: through stream sendrequest体 ( if exists)
173    if let Some(body) = &request.body {
174        if !body.is_empty() {
175            stream
176                .send_data(bytes::Bytes::from(body.clone()))
177                .await
178                .map_err(|e| {
179                    HttpClientError::Http3Error(format!("Failed to send request body: {}", e))
180                })?;
181        }
182    }
183
184    stream
185        .finish()
186        .await
187        .map_err(|e| HttpClientError::Http3Error(format!("completerequestfailure: {}", e)))?;
188
189    // receiveresponse
190    let response = stream
191        .recv_response()
192        .await
193        .map_err(|e| HttpClientError::Http3Error(format!("receiveresponsefailure: {}", e)))?;
194
195    // readresponse体
196    let mut body_data = Vec::new();
197
198    // securitylimit：prevent HTTP/3 responsebody too largecauseinsidememory exhausted
199    const MAX_HTTP3_BODY_SIZE: usize = 100 * 1024 * 1024; // 100MB
200
201    while let Some(mut chunk) = stream.recv_data().await.map_err(|e| {
202        HttpClientError::Io(std::io::Error::other(format!("read body failure: {}", e)))
203    })? {
204        // use Buf trait readcountdata
205        let chunk_len = chunk.remaining();
206
207        // securityCheck：preventresponsebody too large
208        if body_data.len().saturating_add(chunk_len) > MAX_HTTP3_BODY_SIZE {
209            return Err(HttpClientError::InvalidResponse(format!(
210                "HTTP/3 responsebody too large (>{} bytes)",
211                MAX_HTTP3_BODY_SIZE
212            )));
213        }
214
215        let mut chunk_bytes = vec![0u8; chunk_len];
216        chunk.copy_to_slice(&mut chunk_bytes);
217        body_data.extend_from_slice(&chunk_bytes);
218    }
219
220    // Parseresponse
221    let status_code = response.status().as_u16();
222
223    // securityFix: Check HTTP/3 responseheadersize, prevent QPACK compressionbombattack
224    const MAX_HTTP3_HEADER_SIZE: usize = 64 * 1024; // 64KB (RFC 9114 suggestminimumvalue)
225    let total_header_size: usize = response
226        .headers()
227        .iter()
228        .map(|(k, v)| k.as_str().len() + v.len())
229        .sum();
230    if total_header_size > MAX_HTTP3_HEADER_SIZE {
231        return Err(HttpClientError::InvalidResponse(format!(
232            "HTTP/3 responseheadertoo large (>{} bytes)",
233            MAX_HTTP3_HEADER_SIZE
234        )));
235    }
236
237    let status_text = http::StatusCode::from_u16(status_code)
238        .ok()
239        .and_then(|s| s.canonical_reason())
240        .unwrap_or("Unknown")
241        .to_string();
242    let headers = response
243        .headers()
244        .iter()
245        .map(|(k, v)| (k.as_str().to_string(), v.to_str().unwrap_or("").to_string()))
246        .collect();
247
248    Ok(HttpResponse {
249        http_version: "HTTP/3".to_string(),
250        status_code,
251        status_text,
252        headers,
253        body: body_data,
254        response_time_ms: start.elapsed().as_millis() as u64, // 添加实际of响应time测量
255    })
256}

265    #[tokio::test]
266    #[ignore] // neednetworkconnection and HTTP/3 support

268        let user_agent = "TestClient/1.0".to_string();
269        let config = HttpClientConfig {
270            user_agent,
271            prefer_http3: true,
272            ..Default::default()
273        };
274
275        let pool_manager = {
276            #[allow(clippy::arc_with_non_send_sync)]

278                Arc::new(ConnectionPoolManager::new(PoolManagerConfig::default()))

282        let request = HttpRequest::new(HttpMethod::Get, "https://cloudflare-quic.com/");
283
284        let result = send_http3_request_with_pool(
285            "cloudflare-quic.com",
286            443,
287            "/",
288            &request,
289            &config,
290            &pool_manager,
291        )
292        .await;
