52// Fix: use global singleton Runtime avoid frequent Create (for HTTP/2 and HTTP/3 connection pool scenario)
53// Note: only in connection-pool enabled when need, because only connection pool scenario needs sync wrap async code
54#[cfg(all(feature = "connection-pool", any(feature = "http2", feature = "http3")))]
55use once_cell::sync::Lazy;

57#[cfg(all(feature = "connection-pool", any(feature = "http2", feature = "http3")))]
58static SHARED_RUNTIME: Lazy<Result<tokio::runtime::Runtime>> = Lazy::new(|| {
59    tokio::runtime::Runtime::new().map_err(|e| {
60        HttpClientError::ConnectionFailed(format!("Failed to create Tokio runtime: {}", e))
61    })
62});
63
64#[cfg(all(feature = "connection-pool", any(feature = "http2", feature = "http3")))]
65fn get_shared_runtime() -> Result<&'static tokio::runtime::Runtime> {
66    SHARED_RUNTIME.as_ref().map_err(|e| match e {
67        HttpClientError::ConnectionFailed(msg) => HttpClientError::ConnectionFailed(msg.clone()),
68        _ => HttpClientError::ConnectionFailed("Runtime initialization failed".to_string()),
69    })
70}

72/// HTTP client error
73#[derive(Debug)]
74pub enum HttpClientError {
75    Io(std_io::Error),
76    InvalidUrl(String),
77    InvalidResponse(String),
78    TlsError(String),
79    ConnectionFailed(String),
80    Timeout,
81    #[cfg(feature = "http2")]
82    Http2Error(String),
83    #[cfg(feature = "http3")]
84    Http3Error(String),
85    InvalidRequest(String),
86}
87
88impl std::fmt::Display for HttpClientError {
89    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
90        match self {
91            HttpClientError::Io(e) => write!(f, "IO error: {}", e),
92            HttpClientError::InvalidUrl(s) => write!(f, "Invalid URL: {}", s),
93            HttpClientError::InvalidResponse(s) => write!(f, "Invalid response: {}", s),
94            HttpClientError::TlsError(s) => write!(f, "TLS error: {}", s),
95            HttpClientError::ConnectionFailed(s) => write!(f, "Connection failed: {}", s),
96            HttpClientError::Timeout => write!(f, "Request timeout"),
97            #[cfg(feature = "http2")]
98            HttpClientError::Http2Error(s) => write!(f, "HTTP/2 error: {}", s),
99            #[cfg(feature = "http3")]
100            HttpClientError::Http3Error(s) => write!(f, "HTTP/3 error: {}", s),
101            HttpClientError::InvalidRequest(s) => write!(f, "Invalid request: {}", s),
102        }
103    }
104}

108impl From<std_io::Error> for HttpClientError {
109    fn from(err: std_io::Error) -> Self {
110        HttpClientError::Io(err)

242    fn send_request_with_redirects(
243        &self,
244        request: &HttpRequest,
245        redirect_count: usize,
246        request_start: std::time::Instant,
247    ) -> Result<HttpResponse> {

256    /// Inside redirect process (bring loop detect and cumulative timeout protection)
257    fn send_request_with_redirects_internal(
258        &self,
259        request: &HttpRequest,
260        redirect_count: usize,
261        visited_urls: &mut std::collections::HashSet<String>,
262        request_start: std::time::Instant,
263    ) -> Result<HttpResponse> {
264        // Check cumulative timeout (5 minutes maximum for entire request including redirects)
265        if request_start.elapsed() > Duration::from_secs(300) {
266            return Err(HttpClientError::Timeout);
267        }
268
269        // Check redirect times count
270        if redirect_count >= self.config.max_redirects {
271            return Err(HttpClientError::InvalidResponse(format!(
272                "Redirect count exceed limit: {}",
273                self.config.max_redirects
274            )));
275        }
276
277        // Check redirect loop
278        if visited_urls.contains(&request.url) {
279            return Err(HttpClientError::InvalidResponse(format!(
280                "Detect redirect loop: {}",
281                request.url
282            )));
283        }
284        visited_urls.insert(request.url.clone());
285
286        // Parse URL
287        let (scheme, host, port, path) = self.parse_url(&request.url)?;
288
289        // Based on protocol select process method
290        let response = match scheme.as_str() {
291            "http" => self.send_http_request(&host, port, &path, request)?,
292            "https" => self.send_https_request(&host, port, &path, request)?,
293            _ => {
294                return Err(HttpClientError::InvalidUrl(format!(
295                    "Not support protocol: {}",
296                    scheme
297                )));
298            }
299        };
300
301        // Process redirect
302        if (300..400).contains(&response.status_code) {
303            if let Some(location) = response.headers.get("location") {
304                // Build new URL (may is absolute path or relative path)
305                let redirect_url =
306                    if location.starts_with("http://") || location.starts_with("https://") {
307                        location.clone()
308                    } else if location.starts_with("//") {
309                        format!("{}:{}", scheme, location)
310                    } else if location.starts_with('/') {
311                        format!("{}://{}:{}{}", scheme, host, port, location)
312                    } else {
313                        // Relative path
314                        // Fix: correct process path concatenate, avoid double slash
315                        let base_path = if path.ends_with('/') {
316                            &path
317                        } else {
318                            path.rsplit_once('/').map(|(p, _)| p).unwrap_or("/")
319                        };
320                        // Ensure base_path ending with / ending, location not / header
321                        let location = location.trim_start_matches('/');
322                        if base_path == "/" {
323                            format!("{}://{}:{}/{}", scheme, host, port, location)
324                        } else {
325                            format!("{}://{}:{}{}/{}", scheme, host, port, base_path, location)
326                        }
327                    };
328
329                // Fix: Based on HTTP status code correct process redirect method (RFC 7231)
330                let redirect_method = match response.status_code {
331                    301..=303 => {
332                        // 301, 302, 303: POST should change as GET, and remove request body
333                        HttpMethod::Get
334                    }
335                    307 | 308 => {
336                        // 307, 308: keep original HTTP method (POST still is POST)
337                        request.method
338                    }
339                    _ => {
340                        // Other 3xx status code keep original method
341                        request.method
342                    }
343                };
344
345                // Fix: process Set-Cookie (if redirect response has Cookie)
346                if let Some(cookie_store) = &self.config.cookie_store {
347                    if let Some(set_cookie) = response.headers.get("set-cookie") {
348                        // Parse and add Cookie
349                        if let Some(cookie) =
350                            super::cookie::Cookie::parse_set_cookie(set_cookie, host.clone())
351                        {
352                            cookie_store.add_cookie(cookie);
353                        }
354                    }
355                }
356
357                // Parse new URL domain and path (for Cookie field filter)
358                let (new_scheme, new_host, _new_port, new_path) = self.parse_url(&redirect_url)?;
359
360                // Fix: rebuild request, only including suitable for new domain Cookie
361                let mut final_redirect_request = HttpRequest::new(redirect_method, &redirect_url);
362
363                // Copy non Cookie headers, and add Referer
364                for (key, value) in &request.headers {
365                    if key.to_lowercase() != "cookie" {
366                        final_redirect_request = final_redirect_request.with_header(key, value);
367                    }
368                }
369                // Fix: add Referer header (simulate browser behavior)
370                final_redirect_request =
371                    final_redirect_request.with_header("Referer", &request.url);
372
373                // Add suitable for new domain Cookie
374                if let Some(cookie_store) = &self.config.cookie_store {
375                    if let Some(cookie_header) = cookie_store.generate_cookie_header(
376                        &new_host,
377                        &new_path,
378                        new_scheme == "https",
379                    ) {
380                        final_redirect_request =
381                            final_redirect_request.with_header("Cookie", &cookie_header);
382                    }
383                }
384
385                // If keep POST/PUT/PATCH, preserve request body; if change as GET, remove request body (RFC 7231 require)
386                if redirect_method != HttpMethod::Get {
387                    if let Some(body) = &request.body {
388                        final_redirect_request = final_redirect_request.with_body(body.clone());
389                    }
390                }
391
392                // Recursive process redirect (pass visited_urls end with detect loop)
393                return self.send_request_with_redirects_internal(
394                    &final_redirect_request,
395                    redirect_count + 1,
396                    visited_urls,
397                    request_start,
398                );
399            }
400        }
401
402        Ok(response)
403    }

405    /// Parse URL
406    /// Fix: support IPv6 address and correct process query/fragment
407    fn parse_url(&self, url: &str) -> Result<(String, String, u16, String)> {
408        let url = url.trim();
409
410        // Extract scheme
411        let (scheme, rest) = if let Some(stripped) = url.strip_prefix("https://") {
412            ("https", stripped)
413        } else if let Some(stripped) = url.strip_prefix("http://") {
414            ("http", stripped)
415        } else {
416            return Err(HttpClientError::InvalidUrl("Missing protocol".to_string()));
417        };
418
419        // Remove fragment (#后面部分)
420        let rest = if let Some(frag_pos) = rest.find('#') {
421            &rest[..frag_pos]
422        } else {
423            rest
424        };
425
426        // Separate query parameter (?后面部分) and path
427        let (host_port, path_with_query) = if let Some(pos) = rest.find('/') {
428            (&rest[..pos], &rest[pos..])
429        } else {
430            (rest, "/")
431        };
432
433        // Extract path (remove query parameter, but preserve in path in send)
434        // Note: query parameter should preserve in path in, because server need them
435        let path = path_with_query.to_string();
436
437        // Parse host and port
438        // Fix: support IPv6 address format [2001:db8::1]:8080
439        let (host, port) = if host_port.starts_with('[') {
440            // IPv6 address format
441            if let Some(close_bracket) = host_port.find(']') {
442                let host = host_port[1..close_bracket].to_string();
443                if let Some(colon_pos) = host_port[close_bracket + 1..].find(':') {
444                    let port_str = &host_port[close_bracket + 2 + colon_pos..];
445                    let port = port_str
446                        .parse::<u16>()
447                        .map_err(|_| HttpClientError::InvalidUrl("Invalid port".to_string()))?;
448                    (host, port)
449                } else {
450                    let default_port = if scheme == "https" { 443 } else { 80 };
451                    (host, default_port)
452                }
453            } else {
454                return Err(HttpClientError::InvalidUrl(
455                    "IPv6 address format error".to_string(),
456                ));
457            }
458        } else {
459            // IPv4 address or domain
460            if let Some(pos) = host_port.find(':') {
461                let host = host_port[..pos].to_string();
462                let port = host_port[pos + 1..]
463                    .parse::<u16>()
464                    .map_err(|_| HttpClientError::InvalidUrl("Invalid port".to_string()))?;
465                (host, port)
466            } else {
467                let default_port = if scheme == "https" { 443 } else { 80 };
468                (host_port.to_string(), default_port)
469            }
470        };
471
472        Ok((scheme.to_string(), host, port, path))
473    }

475    /// Send HTTP request
476    fn send_http_request(
477        &self,
478        host: &str,
479        port: u16,
480        path: &str,
481        request: &HttpRequest,
482    ) -> Result<HttpResponse> {
483        // If has connection pool, use connection pool
484        #[cfg(feature = "connection-pool")]
485        {
486            if let Some(pool_manager) = &self.pool_manager {
487                return http1_pool::send_http1_request_with_pool(
488                    host,
489                    port,
490                    path,
491                    request,
492                    &self.config,
493                    pool_manager,
494                );
495            }
496        }
497        // Otherwise use ordinary connection
498        http1::send_http1_request(host, port, path, request, &self.config)
499    }

501    /// Send HTTPS request (support HTTP/1.1, HTTP/2, HTTP/3)
502    fn send_https_request(
503        &self,
504        host: &str,
505        port: u16,
506        path: &str,
507        request: &HttpRequest,
508    ) -> Result<HttpResponse> {
509        // If has connection pool, priority use connection pool (HTTPS: HTTP/3 > HTTP/2 > HTTP/1.1)
510        #[cfg(feature = "connection-pool")]
511        if let Some(pool_manager) = &self.pool_manager {
512            // HTTP/3 with pool (async -> sync wrap)
513            #[cfg(feature = "http3")]
514            if self.config.prefer_http3 {
515                // Fix: use global singleton Runtime
516                let runtime = get_shared_runtime()?;
517                return runtime.block_on(async {
518                    http3_pool::send_http3_request_with_pool(
519                        host,
520                        port,
521                        path,
522                        request,
523                        &self.config,
524                        pool_manager,
525                    )
526                    .await
527                });
528            }
529
530            // HTTP/2 with pool (async -> sync wrap)
531            #[cfg(feature = "http2")]
532            if self.config.prefer_http2 {
533                // Fix: use global singleton Runtime
534                // Note: here not do "automatic downgrade", because pool scenario we more hope by user preference go specified protocol
535                // (test also will strict validate version)
536                let runtime = get_shared_runtime()?;
537                return runtime.block_on(async {
538                    http2_pool::send_http2_request_with_pool(
539                        host,
540                        port,
541                        path,
542                        request,
543                        &self.config,
544                        pool_manager,
545                    )
546                    .await
547                });
548            }
549
550            // HTTP/1.1 over TLS with pool
551            return tls::send_https_request_with_pool(
552                host,
553                port,
554                path,
555                request,
556                &self.config,
557                pool_manager,
558            );
559        }
560
561        // Priority: HTTP/3 > HTTP/2 > HTTP/1.1
562
563        // Try HTTP/3
564        #[cfg(feature = "http3")]
565        {
566            if self.config.prefer_http3 {
567                // If opened HTTP/3, we try it.
568                // If failure, we may want to downgrade, but HTTP/3 to TCP is different transfer layer,
569                // Usually if user explicit require HTTP/3, failure then should report error.
570                // But here in order to stable property, if is because protocol error, we can downgrade.
571                // Temporary when keep simple: directly return.
572                match http3::send_http3_request(host, port, path, request, &self.config) {
573                    Ok(resp) => return Ok(resp),
574                    Err(e) => {
575                        // If only only is preference, can try downgrade
576                        // If is Connection failed, may is network issue, also may is server not support
577                        eprintln!("Warning: HTTP/3 failure, try downgrade: {}", e);
578                    }
579                }
580            }
581        }
582
583        // Try HTTP/2
584        #[cfg(feature = "http2")]
585        {
586            if self.config.prefer_http2 {
587                match http2::send_http2_request(host, port, path, request, &self.config) {
588                    Ok(resp) => return Ok(resp),
589                    Err(_e) => {
590                        // Record error but continue try HTTP/1.1
591                        // In actual production should use log system
592                        // eprintln!("HTTP/2 try failure: {}, back to HTTP/1.1", e);
593                    }
594                }
595            }
596        }
597
598        // Back to HTTP/1.1 + TLS
599        tls::send_https_request(host, port, path, request, &self.config)
600    }
