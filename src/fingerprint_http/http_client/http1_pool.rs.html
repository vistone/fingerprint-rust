1//! HTTP/1.1 with Connection Pool
2//!
3//! architectureexplain：
4//! - HTTP/1.1 adopt netconnpool manage TCP connection pool
5//! - pool化pair象：TcpStream (裸 TCP connection)
6//! - reusemethod：serialreuse (anconnectionsame when betweencan onlyprocessanrequest)
7//! - protocollimit：HTTP/1.1 unable tomultiplereuse, needlarge numberconnectionsupportconcurrent
8//! - netconnpool negative责：connectionCreate, keepactive, 故障detect and recycle

18/// useconnection poolsend HTTP/1.1 request
19#[cfg(feature = "connection-pool")]
20pub fn send_http1_request_with_pool(
21    host: &str,
22    port: u16,
23    path: &str,
24    request: &HttpRequest,
25    config: &HttpClientConfig,
26    pool_manager: &Arc<ConnectionPoolManager>,
27) -> Result<HttpResponse> {
28    // from connection poolGetconnection
29    let pool = pool_manager.get_pool(host, port)?;
30
31    // Get TCP connection
32    let conn = pool.get_tcp().map_err(|e| {
33        HttpClientError::ConnectionFailed(format!("Failed to get connection from pool: {:?}", e))
34    })?;
35
36    // from Connection in Extract TcpStream
37    // PooledConnection implement了 Deref<Target = Connection>, candirectlyuse Connection method
38    let tcp_stream = conn.tcp_conn().ok_or_else(|| {
39        HttpClientError::ConnectionFailed("Expected TCP connection but got UDP".to_string())
40    })?;
41
42    // clone TcpStream so thatwecanuse它
43    let mut stream = tcp_stream.try_clone().map_err(HttpClientError::Io)?;
44
45    // Fix: Add Cookie to request ( if exists)
46    let mut request_with_cookies = request.clone();
47    if let Some(cookie_store) = &config.cookie_store {
48        super::request::add_cookies_to_request(
49            &mut request_with_cookies,
50            cookie_store,
51            host,
52            path,
53            false, // HTTP is notsecurityconnection
54        );
55    }
56
57    // Build HTTP request
58    let http_request = request_with_cookies.build_http1_request(host, path);
59
60    // sendrequest
61    stream
62        .write_all(http_request.as_bytes())
63        .map_err(HttpClientError::Io)?;
64
65    // Fix: usecompleteresponsereadlogic (include body)
66    // connectionwillautomatic归still to connection pool (through Drop)
67    let buffer =
68        super::io::read_http1_response_bytes(&mut stream, super::io::DEFAULT_MAX_RESPONSE_BYTES)
69            .map_err(HttpClientError::Io)?;
70
71    // Parseresponse
72    HttpResponse::parse(&buffer).map_err(HttpClientError::InvalidResponse)
73}
